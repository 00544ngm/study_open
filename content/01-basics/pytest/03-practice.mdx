### 练习 1：编写基础测试

```python
# 测试目标函数
def calculate_token_cost(text: str, model: str = "gpt-4") -> float:
    """
    计算文本的 token 费用。
    gpt-4: $0.03/1K tokens
    gpt-3.5-turbo: $0.002/1K tokens
    估算: 1 token ≈ 0.75 个英文词
    """
    word_count = len(text.split())
    estimated_tokens = int(word_count / 0.75)
    rates = {"gpt-4": 0.03, "gpt-3.5-turbo": 0.002}
    rate = rates.get(model, 0.03)
    return (estimated_tokens / 1000) * rate

        pass

        # 测试 gpt-3.5-turbo 计费（更便宜）
        pass
```
