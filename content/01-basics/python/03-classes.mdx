## 核心

### Agent 开发中的类设计模式

```python
from abc import ABC, abstractmethod
from typing import List, Dict, Any

# 抽象基类
class BaseAgent(ABC):
    def __init__(self, name: str, model: str = "gpt-4"):
        self.name = name
        self.model = model
        self.messages: List[Dict[str, str]] = []

    @abstractmethod
    async def run(self, prompt: str) -> str:
        pass

    def add_message(self, role: str, content: str):
        self.messages.append({"role": role, "content": content})

# 具体实现
class ChatAgent(BaseAgent):
    async def run(self, prompt: str) -> str:
        self.add_message("user", prompt)
        # 调用 LLM API
        response = f"模拟回复: {prompt}"
        self.add_message("assistant", response)
        return response

class ToolAgent(BaseAgent):
    def __init__(self, name: str, tools: Dict[str, Any]):
        super().__init__(name)
        self.tools = tools

    async def run(self, prompt: str) -> str:
        # 判断是否需要调用工具
        return f"使用工具处理: {prompt}"
```

## 进阶

### 组合模式（复杂 Agent 系统）

```python
class AgentOrchestrator:
    """管理多个 Agent 的协调器"""
    def __init__(self):
        self.agents: Dict[str, BaseAgent] = {}

    def register_agent(self, name: str, agent: BaseAgent):
        self.agents[name] = agent

    async def route(self, task: str) -> str:
        # 路由到合适的 Agent
        if "search" in task.lower():
            return await self.agents["search"].run(task)
        elif "chat" in task.lower():
            return await self.agents["chat"].run(task)
        return "无法路由任务"
```
