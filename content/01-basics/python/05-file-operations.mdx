## 核心

### Agent 中的文件操作场景

```python
import json
import os
from pathlib import Path
from typing import Any, Dict

class FileMemory:
    """基于文件的长期记忆存储"""
    def __init__(self, storage_dir: str = "memory"):
        self.storage_dir = Path(storage_dir)
        self.storage_dir.mkdir(exist_ok=True)

    def save(self, key: str, data: Dict[str, Any]):
        path = self.storage_dir / f"{key}.json"
        with open(path, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def load(self, key: str) -> Dict[str, Any]:
        path = self.storage_dir / f"{key}.json"
        if not path.exists():
            return {}
        with open(path, "r", encoding="utf-8") as f:
            return json.load(f)

    def list_keys(self) -> list[str]:
        return [f.stem for f in self.storage_dir.glob("*.json")]

# 使用
memory = FileMemory()
memory.save("conversation_001", {"messages": [], "context": {}})
data = memory.load("conversation_001")
```

## 进阶

### 使用 aiofiles 进行异步文件操作

```python
import aiofiles
import asyncio
import json

async def save_conversation(session_id: str, messages: list):
    path = f"conversations/{session_id}.json"
    async with aiofiles.open(path, "w", encoding="utf-8") as f:
        await f.write(json.dumps(messages, ensure_ascii=False))
```
