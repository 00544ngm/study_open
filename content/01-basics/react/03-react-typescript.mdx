## Props 类型定义

### 基础类型

```tsx
interface ButtonProps {
  label: string;
  disabled?: boolean;          // 可选属性
  variant: "primary" | "secondary" | "danger"; // 字面量联合
  size?: "sm" | "md" | "lg";
  onClick: () => void;         // 无参回调
  children?: React.ReactNode;  // 子元素
}

function Button({ label, variant = "primary", size = "md", disabled, onClick, children }: ButtonProps) {
  return (
    <button
      className={`btn btn-${variant} btn-${size}`}
      disabled={disabled}
      onClick={onClick}
    >
      {label}
      {children}
    </button>
  );
}
```

### interface vs type

```tsx
// interface 适合声明对象类型，可合并扩展
interface UserProps {
  name: string;
  email: string;
}
interface UserProps {
  age?: number; // 同名声明的属性合并
}

// type 适合联合、交叉、工具类型
type Status = "idle" | "loading" | "success" | "error";
type Result<T> = { data: T } | { error: string };
type ButtonVariant = "primary" | "secondary";

// Props 通常用 interface（语义更清晰）
// 需要交叉类型时用 type
type ExtendedButtonProps = ButtonProps & {
  tooltip?: string;
  icon?: React.ReactNode;
};
```

## 泛型组件

### 列表与表格组件

```tsx
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string | number;
  emptyState?: React.ReactNode;
}

function List<T>({ items, renderItem, keyExtractor, emptyState }: ListProps<T>) {
  if (items.length === 0) {
    return emptyState ?? <p>暂无数据</p>;
  }

  return (
    <ul>
      {items.map((item, index) => (
        <li key={keyExtractor(item)}>
          {renderItem(item, index)}
        </li>
      ))}
    </ul>
  );
}

// 使用
interface Agent {
  id: string;
  name: string;
  status: "running" | "stopped";
}

<AgentList
  items={agents}
  renderItem={(agent) => (
    <span>{agent.name} — {agent.status}</span>
  )}
  keyExtractor={(agent) => agent.id}
/>
```

### 受控输入组件

```tsx
interface SelectProps<T extends string> {
  value: T;
  options: { label: string; value: T }[];
  onChange: (value: T) => void;
}

function Select<T extends string>({ value, options, onChange }: SelectProps<T>) {
  return (
    <select value={value} onChange={e => onChange(e.target.value as T)}>
      {options.map(opt => (
        <option key={opt.value} value={opt.value}>
          {opt.label}
        </option>
      ))}
    </select>
  );
}

// 使用：类型自动推断
<Select
  value="running"
  options={[
    { label: "运行中", value: "running" },
    { label: "已停止", value: "stopped" },
    { label: "异常", value: "error" },
  ]}
  onChange={(val) => console.log(val)} // val 类型为 "running" | "stopped" | "error"
/>
```

### 多态组件（Polymorphic）

```tsx
type PolymorphicProps<T extends React.ElementType> = {
  as?: T;
  children: React.ReactNode;
} & React.ComponentPropsWithoutRef<T>;

function Box<T extends React.ElementType = "div">({
  as,
  children,
  ...props
}: PolymorphicProps<T>) {
  const Component = as || "div";
  return <Component {...props}>{children}</Component>;
}

// 使用

<Box as="a" href="https://example.com">链接</Box>
```

## 事件处理

### 表单事件

```tsx
interface LoginFormProps {
  onSubmit: (credentials: { username: string; password: string }) => void;
}

function LoginForm({ onSubmit }: LoginFormProps) {
  const [form, setForm] = useState({ username: "", password: "" });

  // 显式类型：React.ChangeEvent<HTMLInputElement>
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target;
    setForm(prev => ({ ...prev, [name]: value }));
  };

  // 显式类型：React.FormEvent<HTMLFormElement>
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    onSubmit(form);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input name="username" value={form.username} onChange={handleChange} />
      <input name="password" type="password" value={form.password} onChange={handleChange} />
      <button type="submit">登录</button>
    </form>
  );
}
```

```tsx
// 自定义事件的类型
interface AgentEventMap {
  "status-change": CustomEvent<{ agentId: string; status: string }>;
  "error": CustomEvent<{ agentId: string; error: Error }>;
  "message": CustomEvent<{ text: string; role: "user" | "assistant" }>;
}

// 类型安全的 EventEmitter
function useAgentEvents(agentId: string) {
  useEffect(() => {
    const handler = (e: Event) => {
      const detail = (e as CustomEvent).detail;
      console.log("Agent 事件:", detail);
    };

    // 类型守卫
    const isStatusChange = (e: Event): e is AgentEventMap["status-change"] =>
      e.type === "status-change";

    window.addEventListener("status-change", handler);
    return () => window.removeEventListener("status-change", handler);
  }, [agentId]);
}
```

## Context 类型

### 创建类型安全的 Context

```tsx
interface ThemeContextValue {
  theme: "light" | "dark";
  toggleTheme: () => void;
}

// 提供默认值（或 undefined + 断言）
const ThemeContext = createContext<ThemeContextValue>({
  theme: "light",
  toggleTheme: () => {},
});

// 使用自定义 Hook 提供类型安全保障
function useTheme(): ThemeContextValue {
  const context = useContext(ThemeContext);
  // 如果创建时用了 undefined 作为默认值，这里需要检查
  if (!context) {
    throw new Error("useTheme 必须在 ThemeProvider 内使用");
  }
  return context;
}
```

### Provider 封装

```tsx
function ThemeProvider({ children }: { children: React.ReactNode }) {
  const [theme, setTheme] = useState<"light" | "dark">("light");

  const toggleTheme = useCallback(() => {
    setTheme(prev => prev === "light" ? "dark" : "light");
  }, []);

  const value = useMemo(
    () => ({ theme, toggleTheme }),
    [theme, toggleTheme]
  );

  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  );
}
```

### Reducer + Context

```tsx
// 定义 Action 类型
type AgentAction =
  | { type: "SET_STATUS"; payload: string }
  | { type: "UPDATE_PROGRESS"; payload: number }
  | { type: "SET_ERROR"; payload: string }
  | { type: "RESET" };

interface AgentState {
  status: "idle" | "running" | "done" | "error";
  progress: number;
  error: string | null;
}

function agentReducer(state: AgentState, action: AgentAction): AgentState {
  switch (action.type) {
    case "SET_STATUS":
      return { ...state, status: action.payload as AgentState["status"] };
    case "UPDATE_PROGRESS":
      return { ...state, progress: action.payload };
    case "SET_ERROR":
      return { ...state, status: "error", error: action.payload };
    case "RESET":
      return { status: "idle", progress: 0, error: null };
    default:
      return state;
  }
}

// 类型安全的 dispatch
const AgentContext = createContext<{
  state: AgentState;
  dispatch: React.Dispatch<AgentAction>;
} | null>(null);

function AgentProvider({ children }: { children: React.ReactNode }) {
  const [state, dispatch] = useReducer(agentReducer, {
    status: "idle",
    progress: 0,
    error: null,
  });

  return (
    <AgentContext.Provider value={{ state, dispatch }}>
      {children}
    </AgentContext.Provider>
  );
}

// 使用
function AgentStatusDisplay() {
  const context = useContext(AgentContext);
  if (!context) throw new Error("必须在 AgentProvider 内使用");

  const { state, dispatch } = context;

  return (
    <div>
      <p>状态: {state.status}</p>
      <p>进度: {state.progress}%</p>
      <button onClick={() => dispatch({ type: "RESET" })}>重置</button>
    </div>
  );
}
```

## Hooks 类型

### useState 与 useReducer

```tsx
// 显式指定泛型参数
const [count, setCount] = useState<number>(0);

// 复杂类型 — TypeScript 通常可推断，但复杂类型建议显式
const [agents, setAgents] = useState<Agent[]>([]);

// useReducer 自动推断 action 类型
const [state, dispatch] = useReducer(agentReducer, initialState);
// dispatch 类型: React.Dispatch<AgentAction>
// 传入不合法 action 时 TypeScript 报错
```

### useRef

```tsx
// DOM 引用：传入初始值 null
const inputRef = useRef<HTMLInputElement>(null);

// 可变值：不涉及 DOM
const intervalRef = useRef<number | null>(null);
intervalRef.current = window.setInterval(() => {}, 1000);

// 保存不需要触发渲染的数据
const cacheRef = useRef<Map<string, Result>>(new Map());

// 需要注意：ref 的 current 是可变的
function focusOnMount() {
  const ref = useRef<HTMLInputElement>(null!); // 非空断言（确定挂载后赋值）
  useEffect(() => { ref.current.focus(); }, []);
  return ref;
}
```

### useImperativeHandle

```tsx
interface ConsoleHandle {
  focus: () => void;
  clear: () => void;
  scrollToBottom: () => void;
}

const AgentConsole = forwardRef<ConsoleHandle, ConsoleProps>(
  function AgentConsole(props, ref) {
    const divRef = useRef<HTMLDivElement>(null);

    useImperativeHandle(ref, () => ({
      focus: () => divRef.current?.focus(),
      clear: () => { /* 清空控制台 */ },
      scrollToBottom: () => {
        divRef.current?.scrollTo(0, divRef.current.scrollHeight);
      },
    }), []);

    return <div ref={divRef} role="log" tabIndex={0} />;
  }
);

// 父组件通过 ref 调用
function Parent() {
  const consoleRef = useRef<ConsoleHandle>(null);
  return (
    <>
      <button onClick={() => consoleRef.current?.clear()}>清空</button>
      <AgentConsole ref={consoleRef} />
    </>
  );
}
```

### 避免常见陷阱

```tsx
// 1. 不要对 props 使用 any
// 错误：interface Props { data: any }
// 正确：interface Props<T> { data: T }

// 2. 事件类型不要用 any
// 错误：onChange={(e: any) => ...}
// 正确：onChange={(e: React.ChangeEvent<HTMLInputElement>) => ...}

// 3. 合理使用非空断言，但不要过度
const user = getUser()!; // 只在确定非 null 时使用

// 4. 使用 satisfies 关键字（TS 4.9+）
const config = {
  variant: "primary",
  size: "md",
} satisfies ButtonProps; // 类型检查但不改变推断类型
```
