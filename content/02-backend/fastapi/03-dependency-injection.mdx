## 入门

```python
from fastapi import Depends, FastAPI

app = FastAPI()

def get_db():
    db = Database()
    try:
        yield db
    finally:
        db.close()

@app.get("/agents")
def list_agents(db: Database = Depends(get_db)):
    return db.query("SELECT * FROM agents")
```

## 核心

```python
from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials

security = HTTPBearer()

# 认证依赖
async def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)):
    token = credentials.credentials
    if not validate_token(token):
        raise HTTPException(status_code=401, detail="Invalid token")
    return token

# 数据库依赖
async def get_db():
    async with async_session() as session:
        yield session

# LLM 客户端依赖
def get_llm_client():
    return LLMClient(api_key=settings.OPENAI_API_KEY)

# 组合使用
@app.post("/api/chat")
async def chat(
    request: ChatRequest,
    token: str = Depends(verify_token),
    db: AsyncSession = Depends(get_db),
    llm: LLMClient = Depends(get_llm_client),
):
    user = await get_user_by_token(token, db)
    response = await llm.chat(request.messages)
    await save_chat_history(db, user.id, request, response)
    return response
```
