## 入门

```sql
-- 连接查询
SELECT u.name, COUNT(s.id) as session_count
FROM users u
LEFT JOIN sessions s ON s.user_id = u.id
GROUP BY u.id
ORDER BY session_count DESC;

-- 窗口函数
SELECT *,
    ROW_NUMBER() OVER (PARTITION BY session_id ORDER BY created_at) as msg_seq
FROM messages;
```

## 核心

```sql
-- 索引优化
CREATE INDEX idx_messages_created_at ON messages(created_at DESC);
CREATE INDEX idx_messages_session_role ON messages(session_id, role);
CREATE INDEX idx_agents_model ON agents(model) WHERE model IS NOT NULL;

-- 查询分析
EXPLAIN ANALYZE
SELECT * FROM messages

-- 全文搜索
ALTER TABLE messages ADD COLUMN search_vector tsvector
GENERATED ALWAYS AS (to_tsvector('simple', content)) STORED;
CREATE INDEX idx_messages_search ON messages USING GIN(search_vector);

SELECT * FROM messages
WHERE search_vector @@ to_tsquery('simple', 'agent & error');
```

```python
# 分页查询（游标分页比 OFFSET 更高效）
async def get_messages_cursor(session_id: str, cursor: int = None, limit: int = 20):
    async with pool.acquire() as conn:
        if cursor:
            rows = await conn.fetch(
                """SELECT * FROM messages
                WHERE session_id = $1 AND id > $2
                ORDER BY id LIMIT $3""",
                session_id, cursor, limit
            )
        else:
            rows = await conn.fetch(
                "SELECT * FROM messages WHERE session_id = $1 ORDER BY id DESC LIMIT $2",
                session_id, limit
            )
        return rows
```

## 进阶

```sql
-- 分区表（按时间）
CREATE TABLE events (
    id BIGSERIAL,
    event_type VARCHAR(50),
    created_at TIMESTAMP NOT NULL
) PARTITION BY RANGE (created_at);

CREATE TABLE events_2025_01
    PARTITION OF events
    FOR VALUES FROM ('2025-01-01') TO ('2025-02-01');

CREATE TABLE events_2025_02
    PARTITION OF events
    FOR VALUES FROM ('2025-02-01') TO ('2025-03-01');

-- 物化视图（预计算统计）
CREATE MATERIALIZED VIEW agent_stats AS
SELECT
    agent_id,
    COUNT(*) as total_calls,
    AVG(token_count) as avg_tokens,
    COUNT(*) FILTER (WHERE role = 'tool') as tool_calls
FROM messages
GROUP BY agent_id;

REFRESH MATERIALIZED VIEW agent_stats;
```
