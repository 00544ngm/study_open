## 核心

### Redis Search

```python
import redis.asyncio as redis

client = redis.Redis(decode_responses=True)

# 创建索引
await client.execute_command(
    "FT.CREATE", "agent-idx",
    "ON", "HASH", "PREFIX", "1", "agent:",
    "SCHEMA",
    "name", "TEXT", "WEIGHT", "5.0",
    "description", "TEXT",
    "model", "TAG",
    "created_at", "NUMERIC", "SORTABLE"
)

# 写入数据
await client.hset("agent:1", mapping={
    "name": "客服助手",
    "description": "处理用户咨询和工单",
    "model": "gpt-4o",
    "created_at": "2025-01-15"
})

# 搜索
result = await client.execute_command(
    "FT.SEARCH", "agent-idx",
    "@name:(客服) @model:{gpt-4o}",
    "RETURN", "3", "name", "model", "description"
)
```

### Redis JSON

```python
import json

# 存储结构化数据
await client.json().set("config:agent-1", "$", {
    "name": "Agent1",
    "tools": ["search", "calculate"],
    "settings": {
        "temperature": 0.7,
        "max_tokens": 2000,
        "retry_count": 3
    },
    "status": "active"
})

# JSONPath 查询
result = await client.json().get("config:agent-1", "$.settings.temperature")
# 返回 [0.7]

# 更新部分字段
await client.json().set("config:agent-1", "$.status", "paused")

# 追加数组
await client.json().arrappend("config:agent-1", "$.tools", "rag")
```

## 进阶

### 限流器

```python
# 滑动窗口限流
class SlidingWindowRateLimiter:
    def __init__(self, client):
        self.client = client

    async def check(self, key: str, limit: int, window: int) -> bool:
        now = int(time.time() * 1000)
        window_start = now - window * 1000

        pipe = self.client.pipeline(transaction=True)
        pipe.zadd(f"ratelimit:{key}", {str(now): now})
        pipe.zremrangebyscore(f"ratelimit:{key}", 0, window_start)
        pipe.zcard(f"ratelimit:{key}")
        pipe.expire(f"ratelimit:{key}", window * 2)
        _, _, count, _ = await pipe.execute()

        return count <= limit

# 使用
limiter = SlidingWindowRateLimiter(client)
if await limiter.check(f"user:{user_id}", limit=10, window=60):
    await process_request()
else:
    raise HTTPException(status_code=429, detail="请求过于频繁")
```
