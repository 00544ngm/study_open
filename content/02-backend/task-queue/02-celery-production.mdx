## 任务路由

### 多队列架构

将不同类型的任务分发到不同队列，避免"慢任务"阻塞"快任务"：

```python
# routing.py
app.conf.task_routes = {
    # 按任务名自动路由
    "tasks.llm.*":          {"queue": "gpu"},       # LLM 调用 → GPU 队列
    "tasks.document.*":     {"queue": "io"},        # 文档处理 → IO 队列
    "tasks.email.*":        {"queue": "urgent"},    # 邮件 → 紧急队列
    "tasks.cleanup.*":      {"queue": "background", "routing_key": "slow"},
}
```

启动 Worker 时按需监听队列：

```bash
# GPU Worker: 只处理 LLM 任务 + 默认任务
celery -A tasks worker -Q gpu,default --concurrency=2

# IO Worker: 处理文档任务（高并发，适合 gevent）
celery -A tasks worker -Q io --concurrency=10 -P gevent

# 紧急队列 Worker: 专门响应
celery -A tasks worker -Q urgent --concurrency=4

# 后台处理 Worker
celery -A tasks worker -Q background --concurrency=1
```

## 高级错误处理

### 指数退避重试

```python
@app.task(
    bind=True,
    max_retries=5,
    default_retry_delay=60,    # 首次重试等待60秒
    autoretry_for=(Exception,),
    retry_backoff=True,         # 启用指数退避
    retry_backoff_max=600,      # 最长等待10分钟
    retry_jitter=True,          # 添加随机抖动避免"惊群"
)
def call_llm_api(self, prompt: str):
    """LLM API 调用含自动退避重试"""
    try:
        response = requests.post(
            "https://api.openai.com/v1/chat/completions",
            json={"model": "gpt-4", "messages": [{"role": "user", "content": prompt}]},
            timeout=30,
        )
        response.raise_for_status()
        return response.json()
    except requests.Timeout:
        # 超时快速重试
        raise self.retry(exc=TimeoutError("API 超时"), countdown=5)
    except requests.HTTPError as e:
        if e.response.status_code == 429:
            # 限流：等待时间从响应头读取
            wait = int(e.response.headers.get("Retry-After", 60))
            raise self.retry(exc=e, countdown=wait)
        elif e.response.status_code >= 500:
            # 服务端错误：指数退避
            raise self.retry(exc=e)
        raise  # 4xx 客户端错误不重试
```

### 死信队列模式

```python
# 重试耗尽后转入死信，而非丢弃
@app.task(
    bind=True,
    max_retries=3,
    routing_key="document",
)
def process_document(self, doc_id: int):
    try:
        doc = fetch_document(doc_id)
        result = analyze_document(doc)
        return result
    except Exception as e:
        if self.request.retries >= self.max_retries - 1:
            # 最后一次重试失败 → 记录到死信队列
            dead_letter.delay(doc_id, str(e), self.request.retries)
        raise self.retry(exc=e, countdown=2 ** self.request.retries * 10)


# 死信处理任务（独立队列，专人排查）
@app.task(queue="dead_letter")
def dead_letter(doc_id: int, error: str, retries: int):
    """记录失败任务到数据库并告警"""
    FailedTask.objects.create(
        task_id=dead_letter.request.id,
        doc_id=doc_id,
        error=error,
        retries=retries,
    )
    # 发送告警
    send_alert(f"文档 {doc_id} 处理失败: {error}")
```

## 签名与画布进阶

### 签名 (Signature)

```python
# 签名是任务 + 参数的封装，可传递、组合、延迟执行
from celery import signature

# 创建签名
process_sig = signature("tasks.process_document", args=[42], kwargs={"lang": "zh"})

# 等同于 .s() 简写语法
process_sig = process_document.s(42, lang="zh")

# 部分参数（partial）
process_sig = process_document.s(doc_id=42)  # 稍后补充其余参数
complete_sig = process_sig.clone(args=(42, "high"), kwargs={"lang": "zh"})
```

### 画布原语 (Canvas)

```python
from celery import chain, group, chord, canvas

# chain — 串联：前一个的输出作为后一个的输入
pipeline = chain(
    fetch_raw_data.s("https://example.com/data"),
    parse_html.s(),
    extract_entities.s(),
    store_to_db.s(),
)
# 等价
pipeline = fetch_raw_data.s("https://example.com/data") | parse_html.s() | extract_entities.s() | store_to_db.s()

result = pipeline()

# group — 并联：所有任务并行执行
parallel = group(
    crawl_page.s("/page/1"),
    crawl_page.s("/page/2"),
    crawl_page.s("/page/3"),
)
results = parallel()

# chord — 并联 + 回调（全部完成后执行回调）
chord(
    group(
        embed_document.s("/docs/doc1.pdf"),
        embed_document.s("/docs/doc2.pdf"),
        embed_document.s("/docs/doc3.pdf"),
    ),
    merge_embeddings.s(),  # 收到所有结果的列表
)()

# 使用 | 操作符的 chord 语法
tasks = group(embed_document.s("/docs/doc1.pdf"),
              embed_document.s("/docs/doc2.pdf")) | merge_embeddings.s()
tasks()
```

### 不可变签名与回调

```python
# 默认情况下 chain 会将前一个结果作为第一个参数传递
# 使用 .si() (immutable signature) 阻止这种行为

@app.task
def notify_admin(message: str):
    """通知管理员（不需要前一步的结果）"""
    send_email("admin@example.com", message)

# 错误：notify_admin 会收到前一步的结果作为额外参数
pipeline = process_document.s(42) | notify_admin.s("处理完成")  
# → notify_admin("处理完成", {"result": ...})  ✓ 正确，extra 参数会被忽略

# 如果不想传递前一步结果，用 immutable
pipeline = process_document.s(42) | notify_admin.si("处理完成")
# → notify_admin("处理完成")

# 回调：在任务完成后执行
from celery.canvas import chain

@app.task(bind=True)
def process_with_callback(self, doc_id):
    """处理文档后自动触发回调"""
    result = do_process(doc_id)

    # 在任务内部创建链式调用
    chain(
        post_process.si(result),
        notify_admin.si(f"文档 {doc_id} 处理完毕"),
    ).apply_async()

    return result
```

## Worker 调优

### 并发模型选择

| 并发模型 | 命令 | 适用场景 | 注意事项 |
|---------|------|---------|---------|
| prefork (默认) | `-P prefork` | CPU 密集型 | 进程隔离，内存占用高 |
| gevent | `-P gevent -c 100` | IO 密集型（网络请求、文件读写） | 需 `monkey.patch_all()` |
| threads | `-P threads -c 20` | IO 密集型 | GIL 限制 CPU 密集 |
| solo | `-P solo` | 调试 | 单线程，阻塞式 |

```python
# 在配置中设定 Worker 默认参数
app.conf.worker_concurrency = 4           # 默认并发数
app.conf.worker_prefetch_multiplier = 1   # 每次只预取1个任务（配合 task_acks_late）
app.conf.worker_max_tasks_per_child = 200 # 每个 Worker 处理200个任务后重启（防内存泄漏）
app.conf.worker_max_memory_per_child = 500000  # 内存超过500MB自动重启
app.conf.worker_send_task_events = True   # 发送任务事件（Flower 需要）
```

# tasks.py — 自定义监控事件
from celery import current_task

@app.task(bind=True)
def monitored_task(self):
    # 发送自定义事件
    self.app.events.Event("myapp-task-started", {
        "task_id": self.request.id,
        "agent_name": self.request.kwargs.get("agent_name"),
    }).send(self.app.events.default_dispatcher)

    try:
        result = perform_work()
        # 上报性能指标
        self.app.events.Event("myapp-performance", {
            "task_id": self.request.id,
            "duration_ms": result.duration_ms,
            "token_count": result.token_count,
        }).send(self.app.events.default_dispatcher)
        return result
    except Exception as e:
        self.app.events.Event("myapp-task-failed", {
            "task_id": self.request.id,
            "error": str(e),
        }).send(self.app.events.default_dispatcher)
        raise

### 优雅关闭

```python
# Worker 信号处理
import signal

@app.task(bind=True)
def long_running_task(self):
    self._stopped = False

    def stop_requested(*args):
        self._stopped = True
        print(f"任务 {self.request.id} 收到停止信号，正在清理...")

    signal.signal(signal.SIGTERM, stop_requested)

    for chunk in stream_data():
        if self._stopped:
            # 保存检查点以便后续恢复
            save_checkpoint(self.request.id, chunk.offset)
            return {"status": "interrupted", "checkpoint": chunk.offset}
        process_chunk(chunk)

    return {"status": "completed"}
```
