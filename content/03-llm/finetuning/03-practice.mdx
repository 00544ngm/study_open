### 练习 3：评估微调效果

```python
# 微调前后对比

def test_model_response(model, tokenizer, prompt: str) -> str:
    """用微调后的模型生成回复。"""
    inputs = tokenizer(prompt, return_tensors="pt").to("cuda")
    outputs = model.generate(
        **inputs,
        max_new_tokens=128,
        temperature=0.1,
    )
    return tokenizer.decode(outputs[0], skip_special_tokens=True)


# 测试用例
test_prompts = [
    "分析代码错误：ZeroDivisionError: division by zero",
    "解释什么是 Python 装饰器",
    "优化这段代码：\nfor i in range(len(arr)):\n    for j in range(len(arr)):\n        if arr[i] < arr[j]:",
]

# 对比指标：
# 1. 回复的相关性（1-5）
# 2. 回复的准确率
# 3. 回复长度是否合适

for prompt in test_prompts:
    response = test_model_response(model, tokenizer, prompt)
    print(f"Prompt: {prompt}")
    print(f"Response: {response}")
    print("---")
```
