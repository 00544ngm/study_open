## 入门

Function Calling 让模型能调用外部工具（API、数据库、搜索引擎等），是构建 Agent 的基础能力。

## 核心

### 定义工具函数

```python
from openai import OpenAI
import json

client = OpenAI()

# 定义工具
tools = [
    {
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "获取指定城市的当前天气",
            "parameters": {
                "type": "object",
                "properties": {
                    "city": {"type": "string", "description": "城市名"},
                    "unit": {"type": "string", "enum": ["celsius", "fahrenheit"]}
                },
                "required": ["city"]
            }
        }
    }
]

# 实际执行函数
def get_weather(city: str, unit: str = "celsius"):
    weather_data = {
        "北京": {"温度": 22, "天气": "晴"},
        "上海": {"温度": 26, "天气": "多云"},
        "广州": {"温度": 30, "天气": "小雨"}
    }
    info = weather_data.get(city, {"温度": 20, "天气": "未知"})
    return json.dumps(info)
```

### 调用流程

```python
messages = [
    {"role": "system", "content": "你是一个天气助手"},
    {"role": "user", "content": "北京今天天气怎么样？"}
]

# 第一步：模型返回工具调用
response = client.chat.completions.create(
    model="gpt-4o",
    messages=messages,
    tools=tools
)

message = response.choices[0].message
print(f"模型决定调用: {message.tool_calls[0].function}")

# 第二步：执行工具
tool_call = message.tool_calls[0]
func_name = tool_call.function.name
args = json.loads(tool_call.function.arguments)

if func_name == "get_weather":
    result = get_weather(**args)

# 第三步：把结果交给模型
messages.append(message)
messages.append({
    "role": "tool",
    "tool_call_id": tool_call.id,
    "content": result
})

final = client.chat.completions.create(
    model="gpt-4o",
    messages=messages,
    tools=tools
)

print(final.choices[0].message.content)
# 北京当前气温22°C，天气晴朗。
```

## 进阶

### 并行工具调用

```python
# 一次调用多个工具
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "对比北京和上海的天气"}],
    tools=tools
)

for tool_call in response.choices[0].message.tool_calls:
    print(f"调用: {tool_call.function.name}({tool_call.function.arguments})")
```

### 带状态的工具执行

```python
class ToolExecutor:
    def __init__(self):
        self.history = []

    def execute(self, tool_calls):
        results = []
        for tc in tool_calls:
            func_name = tc.function.name
            args = json.loads(tc.function.arguments)
            result = self._run(func_name, args)
            self.history.append({"call": func_name, "args": args, "result": result})
            results.append({
                "tool_call_id": tc.id,
                "role": "tool",
                "content": json.dumps(result)
            })
        return results

    def _run(self, name, args):
        if name == "get_weather":
            return {"temperature": 22, "condition": "晴"}
        return {"error": "未知工具"}
```
