## 入门

OpenAI 提供语音转文字（Whisper）和文字转语音（TTS）API，让 Agent 具备语音交互能力。

## 核心

### 语音转文字 (Whisper)

```python
from openai import OpenAI

client = OpenAI()

# 转录音频
with open("recording.mp3", "rb") as f:
    transcript = client.audio.transcriptions.create(
        model="whisper-1",
        file=f,
        response_format="text",  # text / srt / vtt / json
        language="zh",  # 指定语言可提高准确率
    )
print(transcript)

# 翻译成英文（Whisper 的 translate 只支持翻译成英语）
with open("chinese_speech.mp3", "rb") as f:
    translation = client.audio.translations.create(
        model="whisper-1",
        file=f,
    )
print(translation)
```

### 文字转语音 (TTS)

```python
# 生成语音
response = client.audio.speech.create(
    model="tts-1",        # 标准音质
    # model="tts-1-hd",   # 高音质（稍慢）
    voice="alloy",         # alloy / echo / fable / onyx / nova / shimmer
    input="你好，我是你的 AI 助手，有什么可以帮助你的？",
)

# 保存到文件
response.stream_to_file("output.mp3")

# 流式播放
response = client.audio.speech.create(
    model="tts-1",
    voice="alloy",
    input="这是一段较长的语音内容...",
)

for chunk in response.iter_bytes():
    # 边生成边播放
    process_audio_chunk(chunk)
```

### 语音 Agent

```python
import asyncio

class VoiceAgent:
    def __init__(self):
        self.client = OpenAI()

    async def process_voice_input(self, audio_path: str) -> str:
        """语音输入 → 文字 → LLM 处理 → 语音回复"""
        # Step 1: 语音转文字
        with open(audio_path, "rb") as f:
            transcript = self.client.audio.transcriptions.create(
                model="whisper-1", file=f,
            )
        user_text = transcript.text
        print(f"识别: {user_text}")

        # Step 2: LLM 处理
        response = self.client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": user_text}]
        )
        reply = response.choices[0].message.content
        print(f"回复: {reply}")

        # Step 3: 文字转语音
        speech = self.client.audio.speech.create(
            model="tts-1", voice="alloy", input=reply
        )

        output_path = "reply.mp3"
        speech.stream_to_file(output_path)
        return output_path
```

### Voice + Tool Calling

```python
# 语音控制 Agent：用户说话，Agent 执行工具
voice_tools = [
    {
        "type": "function",
        "function": {
            "name": "set_reminder",
            "description": "设置提醒",
            "parameters": {"type": "object", "properties": {
                "time": {"type": "string"},
                "content": {"type": "string"}
            }, "required": ["time", "content"]}
        }
    }
]

def voice_command_pipeline(audio_path: str):
    client = OpenAI()

    # 语音 → 文字
    with open(audio_path, "rb") as f:
        text = client.audio.transcriptions.create(
            model="whisper-1", file=f,
        ).text

    # 文字 → 工具调用
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": text}],
        tools=voice_tools,
    )

    return response.choices[0].message
```
