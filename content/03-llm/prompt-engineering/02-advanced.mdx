## 入门

进阶技巧让模型输出更可控、更准确。

## 核心

### Few-Shot Prompting

```python
from openai import OpenAI

client = OpenAI()

# 给模型示范几个例子
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "将用户输入分类为: positive / negative / neutral"},
        {"role": "user", "content": "这部电影太棒了！"},
        {"role": "assistant", "content": "positive"},
        {"role": "user", "content": "一般般吧，凑合看"},
        {"role": "assistant", "content": "neutral"},
        {"role": "user", "content": "今天天气真好"},
        # 模型会根据前面的示例推断
    ]
)
print(response.choices[0].message.content)
```

### Chain-of-Thought (CoT)

让模型逐步推理，提升复杂问题正确率。

```python
response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "请逐步推理，最后给出答案。"},
        {"role": "user", "content": """一个池塘里的荷花每天翻倍增长。
第30天覆盖整个池塘。问：第几天覆盖一半？
请一步步思考。"""}
    ]
)
print(response.choices[0].message.content)
# 输出会包含推理过程: 第29天覆盖一半 → 第30天翻倍全覆盖
```

### 结构化输出

```python
from pydantic import BaseModel

class ExtractInfo(BaseModel):
    name: str
    age: int
    skills: list[str]

response = client.beta.chat.completions.parse(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "从文本中提取结构化信息"},
        {"role": "user", "content": "张三，28岁，擅长Python、Go和Kubernetes"}
    ],
    response_format=ExtractInfo
)

info = response.choices[0].message.parsed
print(info.name, info.age, info.skills)
# 张三 28 ['Python', 'Go', 'Kubernetes']
```

## 进阶

### 思维树 (Tree-of-Thought)

```python
# 多路径推理，选择最优解
import itertools

def tree_of_thought(prompt, branches=3, depth=2):
    """简单 ToT 实现：多路径探索 + 投票选择"""
    current = [prompt]

    for level in range(depth):
        # 每条路径分出多个分支
        candidates = []
        for path in current:
            response = client.chat.completions.create(
                model="gpt-4o",
                messages=[
                    {"role": "system", "content": f"为当前思路想{branches}种不同的下一步"},
                    {"role": "user", "content": path}
                ]
            )
            candidates.append(response.choices[0].message.content)

        # 投票选择最优
        vote_prompt = f"从以下思路中选择最好的:\n" + \
                      "\n".join(f"{i}. {c}" for i, c in enumerate(candidates))
        vote = client.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": vote_prompt}]
        )
        current = [vote.choices[0].message.content]

    return current[0]
```

### ReAct (Reasoning + Acting)

```python
# 交替推理和行动，适用于 Agent 场景
def react_agent(question: str, max_steps=5):
    messages = [
        {"role": "system", "content": """你是一个能思考并行动的Agent。
每次回复格式：
思考: <当前推理>
行动: <需要执行的操作>
---"""},
        {"role": "user", "content": question}
    ]

    for _ in range(max_steps):
        response = client.chat.completions.create(
            model="gpt-4o", messages=messages
        )
        reply = response.choices[0].message.content
        messages.append({"role": "assistant", "content": reply})

        if "最终答案" in reply:
            return reply

        # 模拟执行 Action（实际应调用工具）
        action_result = f"观察到结果: {reply} 执行完毕"
        messages.append({"role": "user", "content": action_result})

    return "达到最大步数"

print(react_agent("帮我查一下北京今天的天气，然后告诉我该穿什么"))
```
