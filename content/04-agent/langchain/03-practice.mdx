### 练习 1：Chain 和 Prompt 模板

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate
from langchain.schema import SystemMessage, HumanMessage

# 1. 创建 Prompt 模板
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是{role}专家。请用{language}回答。"),
    ("human", "{question}"),
])

# 2. 创建 LLM
llm = ChatOpenAI(model="gpt-4o", temperature=0.1)

# 3. 组合 Chain
chain = prompt | llm

# 4. 调用 Chain
result = chain.invoke({
    "role": "Python",
    "language": "中文",
    "question": "什么是装饰器？",
})
print(result.content)
```

### 练习 2：自定义工具

```python
from langchain.tools import tool


@tool
def get_current_time() -> str:
    """返回当前时间。"""
    from datetime import datetime
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")


@tool
def search_knowledge_base(query: str) -> str:
    """搜索知识库，query 是搜索关键词。"""
    # 模拟知识库
    knowledge = {
        "python": "Python 是一种动态类型的编程语言",
        "agent": "AI Agent 是能自主执行任务的智能体",
        "langchain": "LangChain 是 LLM 应用开发框架",
    }
    return knowledge.get(query.lower(), f"未找到 {query} 的相关信息")


# 测试工具
print(get_current_time.name)  # 应输出: get_current_time
print(search_knowledge_base("agent"))
```

### 练习 3：创建 Agent 执行器

```python
from langchain.agents import create_react_agent, AgentExecutor
from langchain_openai import ChatOpenAI

# 1. 准备工具
tools = [get_current_time, search_knowledge_base]

# 2. 创建 LLM
llm = ChatOpenAI(model="gpt-4o", temperature=0)

# 3. 创建 ReAct Agent
from langchain.agents import Tool
from langchain.prompts import PromptTemplate

# 标准 ReAct prompt
react_prompt = PromptTemplate.from_template(
    "你是一个有用的 AI 助手。\n\n"
    "可用工具：\n{tools}\n\n"
    "工具名称：{tool_names}\n\n"
    "用户输入：{input}\n\n"
    "请逐步思考并选择工具。\n{agent_scratchpad}"
)

agent = create_react_agent(llm, tools, react_prompt)
agent_executor = AgentExecutor(agent=agent, tools=tools, verbose=True)

# 4. 运行 Agent
result = agent_executor.invoke({"input": "现在几点？用中文回答"})
print(result["output"])
```
