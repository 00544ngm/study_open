## 概述

这是本系列的**集大成项目**。你将构建一个生产级的 **AI 文档问答系统**，串联起之前学过的所有阶段：

- **Stage 2 (FastAPI)**：REST API + WebSocket + PostgreSQL
- **Stage 4 (LangGraph)**：Agent 状态图编排
- **Stage 6 (RAG)**：Qdrant 向量库 + 混合检索
- **Stage 5 (Monitoring)**：LangSmith 追踪 + 指标采集
- **Stage 8 (Docker)**：容器化部署 + Nginx

```
用户上传 PDF → 文档解析 → 文本分块 → 向量化 → Qdrant 存储
                                                    ↓
用户提问 → FastAPI → LangGraph Agent → RAG 检索 → LLM 生成 → 流式返回
             ↓                ↓                 ↓
        PostgreSQL       LangSmith          Token 统计
        (对话历史)      (追踪调试)         (用量监控)
```

```python
# app/db_models.py
from sqlalchemy import Column, Integer, String, Text, DateTime, ForeignKey, JSON
from sqlalchemy.orm import DeclarativeBase, relationship
from datetime import datetime, timezone


class Base(DeclarativeBase):
    pass


class Conversation(Base):
    __tablename__ = "conversations"

    id = Column(Integer, primary_key=True, autoincrement=True)
    session_id = Column(String(64), unique=True, nullable=False, index=True)
    title = Column(String(255), default="新对话")
    created_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))
    updated_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc), onupdate=lambda: datetime.now(timezone.utc))

    messages = relationship("Message", back_populates="conversation", cascade="all, delete-orphan", order_by="Message.created_at")


class Message(Base):
    __tablename__ = "messages"

    id = Column(Integer, primary_key=True, autoincrement=True)
    conversation_id = Column(Integer, ForeignKey("conversations.id"), nullable=False)
    role = Column(String(16), nullable=False)  # user / assistant / system / tool
    content = Column(Text, nullable=False)
    metadata_json = Column(JSON, nullable=True)  # 额外信息（token 数、延迟等）
    created_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))

    conversation = relationship("Conversation", back_populates="messages")


class Document(Base):
    __tablename__ = "documents"

    id = Column(Integer, primary_key=True, autoincrement=True)
    filename = Column(String(255), nullable=False)
    file_path = Column(String(512), nullable=False)
    file_size = Column(Integer, nullable=False)  # 字节数
    chunk_count = Column(Integer, default=0)
    status = Column(String(16), default="processing")  # processing / indexed / failed
    created_at = Column(DateTime(timezone=True), default=lambda: datetime.now(timezone.utc))
```

```python
# app/database.py
from sqlalchemy.ext.asyncio import create_async_engine, async_sessionmaker, AsyncSession
from app.config import get_settings
from app.db_models import Base

settings = get_settings()

engine = create_async_engine(
    settings.database_url,
    echo=settings.debug,
    pool_size=10,
    max_overflow=20,
)

AsyncSessionLocal = async_sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)


async def init_db():
    """初始化数据库表结构"""
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)


async def get_session() -> AsyncSession:
    """FastAPI 依赖注入：获取数据库会话"""
    async with AsyncSessionLocal() as session:
        try:
            yield session
        finally:
            await session.close()
```

```python
# app/routes/chat.py
import json
import uuid
from datetime import datetime, timezone

from fastapi import APIRouter, Depends, HTTPException, WebSocket, WebSocketDisconnect
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from sse_starlette.sse import EventSourceResponse

from app.database import get_session
from app.models import ChatRequest, ChatResponse
from app.db_models import Conversation, Message
from app.agent.graph import build_qa_graph
from app.config import get_settings

router = APIRouter()


@router.post("/chat", response_model=ChatResponse)
async def chat(
    req: ChatRequest,
    session: AsyncSession = Depends(get_session),
):
    """REST 接口：发送消息并获取回复"""
    settings = get_settings()
    session_id = req.session_id or str(uuid.uuid4())

    # 获取或创建会话
    result = await session.execute(
        select(Conversation).where(Conversation.session_id == session_id)
    )
    conv = result.scalar_one_or_none()
    if not conv:
        conv = Conversation(session_id=session_id, title=req.message[:50])
        session.add(conv)
        await session.commit()

    # 保存用户消息
    user_msg = Message(
        conversation_id=conv.id,
        role="user",
        content=req.message,
    )
    session.add(user_msg)

    # 构建并运行 Agent
    graph = build_qa_graph()
    config = {
        "configurable": {"thread_id": session_id},
        "metadata": {"session_id": session_id},
    }

    try:
        result = await graph.ainvoke(
            {
                "messages": [{"role": "user", "content": req.message}],
                "session_id": session_id,
                "retrieved_docs": [],
                "sources": [],
                "token_usage": {},
            },
            config,
        )

        reply = result["messages"][-1]["content"]
        sources = result.get("sources", [])
        token_usage = result.get("token_usage", {})

        # 保存助手回复
        assistant_msg = Message(
            conversation_id=conv.id,
            role="assistant",
            content=reply,
            metadata_json={
                "sources": sources,
                "token_usage": token_usage,
            },
        )
        session.add(assistant_msg)

        # 更新会话标题（如果是第一条消息）
        if conv.title == "新对话" or conv.title == req.message[:50]:
            messages_count = await _count_messages(session, conv.id)
            if messages_count <= 2:
                conv.title = req.message[:50]

        conv.updated_at = datetime.now(timezone.utc)
        await session.commit()

        return ChatResponse(
            reply=reply,
            session_id=session_id,
            sources=sources,
            token_usage=token_usage,
        )

    except Exception as e:
        await session.rollback()
        raise HTTPException(status_code=500, detail=f"Agent 处理失败: {str(e)}")
```

### 5.3 流式对话路由（SSE）

```python
# 续：app/routes/chat.py

@router.post("/chat/stream")
async def chat_stream(
    req: ChatRequest,
    session: AsyncSession = Depends(get_session),
):
    """SSE 流式接口：逐字返回 Agent 推理过程和回答"""
    settings = get_settings()
    session_id = req.session_id or str(uuid.uuid4())

    # 获取或创建会话
    result = await session.execute(
        select(Conversation).where(Conversation.session_id == session_id)
    )
    conv = result.scalar_one_or_none()
    if not conv:
        conv = Conversation(session_id=session_id, title=req.message[:50])
        session.add(conv)
        await session.commit()

    user_msg = Message(
        conversation_id=conv.id,
        role="user",
        content=req.message,
    )
    session.add(user_msg)
    await session.commit()

    async def event_generator():
        graph = build_qa_graph()
        config = {
            "configurable": {"thread_id": session_id},
            "metadata": {"session_id": session_id},
        }

        try:
            # 使用 astream_events 获取流式事件
            async for event in graph.astream_events(
                {
                    "messages": [{"role": "user", "content": req.message}],
                    "session_id": session_id,
                    "retrieved_docs": [],
                    "sources": [],
                    "token_usage": {},
                },
                config,
                version="v1",
            ):
                kind = event["event"]

                if kind == "on_chat_model_stream":
                    chunk = event["data"]["chunk"]
                    if chunk and chunk.content:
                        yield {
                            "event": "token",
                            "data": json.dumps({"token": chunk.content}, ensure_ascii=False),
                        }

                elif kind == "on_tool_start":
                    yield {
                        "event": "tool_start",
                        "data": json.dumps({
                            "tool": event["name"],
                            "input": event["data"].get("input", {}),
                        }, ensure_ascii=False),
                    }

                elif kind == "on_tool_end":
                    yield {
                        "event": "tool_end",
                        "data": json.dumps({
                            "tool": event["name"],
                            "output": str(event["data"].get("output", ""))[:200],
                        }, ensure_ascii=False),
                    }

            # 完成信号
            yield {
                "event": "done",
                "data": json.dumps({"session_id": session_id}, ensure_ascii=False),
            }

            # 最终状态获取 sources 和 token 用量
            final_state = await graph.aget_state(config)
            if final_state and final_state.values:
                sources = final_state.values.get("sources", [])
                token_usage = final_state.values.get("token_usage", {})
                yield {
                    "event": "metadata",
                    "data": json.dumps({
                        "sources": sources,
                        "token_usage": token_usage,
                    }, ensure_ascii=False),
                }

                # 保存助手回复到数据库
                messages = final_state.values.get("messages", [])
                if messages:
                    last_content = messages[-1]["content"]
                    assistant_msg = Message(
                        conversation_id=conv.id,
                        role="assistant",
                        content=last_content,
                        metadata_json={"sources": sources, "token_usage": token_usage},
                    )
                    session.add(assistant_msg)
                    await session.commit()

        except Exception as e:
            yield {
                "event": "error",
                "data": json.dumps({"error": str(e)}, ensure_ascii=False),
            }

    return EventSourceResponse(event_generator())
```

### 5.4 WebSocket 对话路由

```python
# 续：app/routes/chat.py

@router.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    """WebSocket 接口：全双工流式对话"""
    await websocket.accept()
    settings = get_settings()
    session_id = str(uuid.uuid4())
    graph = build_qa_graph()

    try:
        while True:
            data = await websocket.receive_text()
            request_data = json.loads(data)
            message = request_data.get("message", "")
            if request_data.get("session_id"):
                session_id = request_data["session_id"]

            config = {
                "configurable": {"thread_id": session_id},
                "metadata": {"session_id": session_id},
            }

            full_reply = ""
            async for event in graph.astream_events(
                {
                    "messages": [{"role": "user", "content": message}],
                    "session_id": session_id,
                    "retrieved_docs": [],
                    "sources": [],
                    "token_usage": {},
                },
                config,
                version="v1",
            ):
                if event["event"] == "on_chat_model_stream":
                    chunk = event["data"]["chunk"]
                    if chunk and chunk.content:
                        full_reply += chunk.content
                        await websocket.send_json({
                            "type": "token",
                            "content": chunk.content,
                        })

                elif event["event"] == "on_tool_start":
                    await websocket.send_json({
                        "type": "tool_start",
                        "tool": event["name"],
                        "input": str(event["data"].get("input", {})),
                    })

                elif event["event"] == "on_tool_end":
                    await websocket.send_json({
                        "type": "tool_end",
                        "tool": event["name"],
                        "output": str(event["data"].get("output", ""))[:200],
                    })

            final_state = await graph.aget_state(config)
            sources = final_state.values.get("sources", []) if final_state else []
            await websocket.send_json({
                "type": "done",
                "session_id": session_id,
                "sources": sources,
```

```python
# app/rag/ingestion.py
import os
from pathlib import Path
from typing import Optional

from pypdf import PdfReader
from langchain.text_splitter import RecursiveCharacterTextSplitter

from app.rag.embedding import get_embedding_function
from app.rag.retriever import get_qdrant_client, get_qdrant_collection


def extract_text_from_pdf(file_path: str) -> str:
    """从 PDF 文件中提取文本"""
    reader = PdfReader(file_path)
    text_parts = []

    for page_num, page in enumerate(reader.pages, start=1):
        page_text = page.extract_text()
        if page_text.strip():
            text_parts.append(f"--- 第 {page_num} 页 ---\n{page_text}")

    return "\n\n".join(text_parts)


def chunk_text(text: str, chunk_size: int = 512, chunk_overlap: int = 128) -> list[str]:
    """递归字符文本分割器 — 保持段落和句子完整性"""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=chunk_size,
        chunk_overlap=chunk_overlap,
        separators=["\n\n", "\n", "。", ".", " ", ""],
        length_function=len,
    )
    chunks = splitter.split_text(text)
    return chunks


async def process_document(file_path: str, document_id: int) -> int:
    """
    完整处理流程：
    1. 提取 PDF 文本
    2. 分块
    3. 向量化并存储到 Qdrant
    """
    # Step 1: 提取文本
    text = extract_text_from_pdf(file_path)
    if not text.strip():
        raise ValueError("PDF 文件中未提取到任何文本内容")

    # Step 2: 分块
    chunks = chunk_text(text)
    print(f"[索引] 文档 {document_id}: 共 {len(chunks)} 个文本块")

    # Step 3: 向量化与存储
    embedding_fn = get_embedding_function()
    embeddings = embedding_fn.embed_documents(chunks)

    client = get_qdrant_client()
    collection_name = get_qdrant_collection()

    # 准备 Qdrant 点数据
    from qdrant_client.models import PointStruct
    import uuid

    points = []
    for i, (chunk, embedding) in enumerate(zip(chunks, embeddings)):
        point_id = str(uuid.uuid4())
        points.append(PointStruct(
            id=point_id,
            vector=embedding,
            payload={
                "document_id": document_id,
                "chunk_index": i,
                "text": chunk,
                "source_file": Path(file_path).name,
            },
        ))

    # 批量上传（每批 100 条）
    batch_size = 100
    for i in range(0, len(points), batch_size):
        batch = points[i : i + batch_size]
        client.upsert(
            collection_name=collection_name,
            points=batch,
        )
        print(f"[索引] 已上传第 {i + 1}-{min(i + batch_size, len(points))} 条向量")

    return len(chunks)
```

```python
def ensure_collection_exists():
    """确保集合存在，不存在则创建"""
    client = get_qdrant_client()
    collection_name = get_qdrant_collection()

    collections = client.get_collections().collections
    existing = [c.name for c in collections]

    if collection_name not in existing:
        from qdrant_client.models import VectorParams, Distance, SparseVectorParams, SparseIndexParams

        client.create_collection(
            collection_name=collection_name,
            vectors_config=VectorParams(
                size=get_embedding_dimension(),
                distance=Distance.COSINE,
            ),
            sparse_vectors_config={
                "sparse": SparseVectorParams(
                    index=SparseIndexParams(
                        on_disk=False,
                    )
                ),
            },
        )
        print(f"[Qdrant] 已创建集合: {collection_name}")


async def hybrid_search(
    query: str,
    document_id: Optional[int] = None,
    top_k: int = 5,
) -> list[dict]:
    """
    混合检索：密集向量 + 稀疏关键词
    返回 [{text, score, source, document_id, chunk_index}]
    """
    client = get_qdrant_client()
    collection_name = get_qdrant_collection()
    embedding_fn = get_embedding_function()

    # 生成查询向量
    query_vector = embedding_fn.embed_query(query)

    # 构建过滤条件
    query_filter = None
    if document_id is not None:
        query_filter = Filter(
            must=[
                FieldCondition(
                    key="document_id",
                    match=MatchValue(value=document_id),
                )
            ]
        )

    # 执行搜索
    search_result = client.query_points(
        collection_name=collection_name,
        query=query_vector,
        query_filter=query_filter,
        limit=top_k,
        with_payload=True,
        search_params=SearchParams(
            exact=False,
            hnsw_ef=128,
        ),
    )

    results = []
    for point in search_result.points:
        payload = point.payload or {}
        results.append({
            "text": payload.get("text", ""),
            "score": point.score,
            "source": payload.get("source_file", ""),
            "document_id": payload.get("document_id"),
            "chunk_index": payload.get("chunk_index"),
        })

    return results


async def dense_search(
    query: str,
    document_id: Optional[int] = None,

    embedding_fn = get_embedding_function()

    query_vector = embedding_fn.embed_query(query)

    query_filter = None
    if document_id is not None:
        query_filter = Filter(
            must=[
                FieldCondition(
                    key="document_id",
                    match=MatchValue(value=document_id),
                )
            ]
        )

    search_result = client.search(
        collection_name=collection_name,
        query_vector=query_vector,
        query_filter=query_filter,
        limit=top_k,
        with_payload=True,
    )

    results = []
    for point in search_result:
        payload = point.payload or {}
        results.append({
            "text": payload.get("text", ""),
            "score": point.score,
            "source": payload.get("source_file", ""),
            "document_id": payload.get("document_id"),
            "chunk_index": payload.get("chunk_index"),
        })

    return results
```

```python
        return "web_search"
    else:
        return "retrieve_documents"  # 默认走 RAG


def route_after_retrieve(state: AgentState) -> str:
    """检索后判断是否需要补充联网搜索"""
    docs = state.get("retrieved_docs", [])
    if not docs or docs == ["未找到相关文档内容。"]:
        return "web_search"  # 本地检索为空，尝试联网
    return "generate_response"


def route_after_generate(state: AgentState) -> str:
    """生成后是否进行格式化后处理"""
    # 有来源引用时走格式化
    sources = state.get("sources", [])
    if sources:
        return "format_response"
    return END


def route_after_web_search(state: AgentState) -> str:
    """联网搜索后生成回答"""
    return "generate_response"


def build_qa_graph():
    """构建完整的问答状态图"""
    workflow = StateGraph(AgentState)

    # 注册所有节点
    workflow.add_node("classify_intent", classify_intent)
    workflow.add_node("retrieve_documents", retrieve_documents)
    workflow.add_node("generate_response", generate_response)
    workflow.add_node("web_search", web_search_node)
    workflow.add_node("format_response", format_response)

    # 设置入口
    workflow.set_entry_point("classify_intent")

    # 分类 → 条件路由
    workflow.add_conditional_edges(
        "classify_intent",
        route_after_classify,
        {
            "retrieve_documents": "retrieve_documents",
            "web_search": "web_search",
            "generate_response": "generate_response",
        },
    )

    # 检索 → 条件路由（本地检索不到则联网）
    workflow.add_conditional_edges(
        "retrieve_documents",
        route_after_retrieve,
        {
            "generate_response": "generate_response",
            "web_search": "web_search",
        },
    )

    # 联网 → 生成
    workflow.add_edge("web_search", "generate_response")

    # 生成 → 条件路由（有来源则格式化）
    workflow.add_conditional_edges(
        "generate_response",
        route_after_generate,
        {
            "format_response": "format_response",
            END: END,
        },
    )

    # 格式化 → 结束
    workflow.add_edge("format_response", END)

    # 编译图（带 SQLite Checkpointer）
    checkpointer = SqliteSaver.from_conn_string("langgraph_checkpoints.db")

    graph = workflow.compile(
        checkpointer=checkpointer,
    )

    return graph
```

```python
def get_token_usage_callback():
    """
    返回一个简单的 token 用量追踪回调。
    在实际项目中集成 langchain.callbacks 使用。
    """
    from langchain.callbacks.base import BaseCallbackHandler

    class TokenUsageHandler(BaseCallbackHandler):
        """追踪每次 LLM 调用的 token 用量"""

        def __init__(self):
            self.total_prompt_tokens = 0
            self.total_completion_tokens = 0
            self.total_cost = 0.0
            self.call_count = 0

        def on_llm_end(self, response, **kwargs):
            """LLM 调用结束时记录用量"""
            try:
                usage = response.llm_output or {}
                token_usage = usage.get("token_usage", {})
                prompt_tokens = token_usage.get("prompt_tokens", 0)
                completion_tokens = token_usage.get("completion_tokens", 0)

                self.total_prompt_tokens += prompt_tokens
                self.total_completion_tokens += completion_tokens
                self.call_count += 1

                # GPT-4o 大致价格（每 1K token）
                # 输入: $0.0025 / 1K tokens
                # 输出: $0.01 / 1K tokens
                cost = (prompt_tokens * 0.0025 + completion_tokens * 0.01) / 1000
                self.total_cost += cost

            except Exception:
                pass

        def get_summary(self) -> dict:
            return {
                "llm_calls": self.call_count,
                "prompt_tokens": self.total_prompt_tokens,
                "completion_tokens": self.total_completion_tokens,
                "total_tokens": self.total_prompt_tokens + self.total_completion_tokens,
                "estimated_cost_usd": round(self.total_cost, 6),
            }

    return TokenUsageHandler()
```

## 扩展方向

| 功能 | 实现方式 |
|------|---------|
| **多租户** | 在 Document payload 中添加 tenant_id 字段，检索时按 Filter 隔离 |
| **多模态 RAG** | 集成 VLMs（如 GPT-4o Vision），支持图片/表格/图表问答 |
| **Agent 记忆** | 使用 LangGraph PostgresSaver 替代 SqliteSaver，支持跨会话记忆 |
| **定时索引** | 集成 Celery Beat 定时扫描文件目录，自动增量更新 Qdrant |
| **用户反馈** | 添加 thumbs up/down API，反馈数据用于 LangSmith 标注和微调 |
| **Guarded 输出** | 添加 Guardrails 节点，检测有害内容、幻觉和 PII 泄露 |
| **GraphRAG** | 使用知识图谱增强 RAG，支持多跳推理和关系查询 |
| **A/B 评测** | LangSmith 数据集 + 自动化评估 Pipeline，对比不同模型/参数 |

## 小结

- 本项目串联了本课程的所有核心阶段：**FastAPI (Stage 2) + LangGraph (Stage 4) + RAG (Stage 6) + 监控 (Stage 5) + Docker (Stage 8)**
- **LangGraph StateGraph** 将对话流程建模为有向图，支持条件路由、多步推理和 Checkpointer 持久化
- **RAG Pipeline** 使用 Qdrant 混合检索（密集向量 + 稀疏关键词），通过 RecursiveCharacterTextSplitter 保持文本语义完整性
- **流式传输** 支持 REST、SSE 和 WebSocket 三种方式，适应不同前端场景
- **LangSmith** 提供端到端可观测性，便于调试和优化 Agent 行为
- **Docker Compose** 一键编排 5 个服务（API / PostgreSQL / Redis / Qdrant / Nginx），生产就绪
- 全英文代码+中文注释的设计保持代码通用性，同时让中文开发者无障碍理解
