import pytest


# 被测试的函数
async def get_agent_response(prompt: str) -> str:
    """调用 LLM 获取回复。"""
    from langchain_openai import ChatOpenAI
    llm = ChatOpenAI(model="gpt-4")
    response = await llm.ainvoke(prompt)
    return response.content


@pytest.mark.asyncio
async def test_get_agent_response():
    """使用 Mock 模拟 LLM 调用。"""
    # 1. 用 patch 模拟 ChatOpenAI
    with patch("langchain_openai.ChatOpenAI") as MockLLM:
        mock_instance = MockLLM.return_value
        # 配置 mock 返回值
        mock_response = AsyncMock()
        mock_response.content = "这是模拟的回复"
        mock_instance.ainvoke.return_value = mock_response

        # 2. 调用被测试函数
        result = await get_agent_response("你好")

        # 3. 验证
        assert result == "这是模拟的回复"
        mock_instance.ainvoke.assert_called_once_with("你好")
        print("测试通过！")

class AgentState:
    """管理 Agent 的运行状态。"""

    def __init__(self):
        self.status = "idle"  # idle | running | paused | error
        self.tokens_used = 0
        self.errors: list[str] = []

    def start(self):
        if self.status == "running":
            raise RuntimeError("Agent already running")
        self.status = "running"

    def pause(self):
        if self.status != "running":
            raise RuntimeError("Can only pause a running agent")
        self.status = "paused"

    def error(self, msg: str):
        self.status = "error"
        self.errors.append(msg)

    def add_tokens(self, count: int):
        if count < 0:
            raise ValueError("Token count cannot be negative")
        self.tokens_used += count
