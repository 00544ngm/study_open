### 练习 2：流式输出处理

```python
import asyncio


class StreamBuffer:
    """
    处理流式输出，支持逐字符/逐块追加。
    支持中断信号，让 UI 能停止流式渲染。
    """

    def __init__(self):
        self.buffer = ""
        self._stopped = False

    def append(self, chunk: str):
        """追加一个数据块。"""
        if not self._stopped:
            self.buffer += chunk

    def stop(self):
        """停止接收数据。"""
        self._stopped = True

    @property
    def is_stopped(self) -> bool:
        return self._stopped

    def clear(self):
        """清空缓冲区。"""
        self.buffer = ""
        self._stopped = False


async def simulate_stream(output: str, buffer: StreamBuffer, delay: float = 0.02):
    """模拟流式输出，逐字符写入 buffer。"""
    for char in output:
        if buffer.is_stopped:
            break
        buffer.append(char)
        await asyncio.sleep(delay)


# 测试
async def test_stream():
    buffer = StreamBuffer()
    task = asyncio.create_task(simulate_stream("Hello, World!", buffer))

    await asyncio.sleep(0.1)  # 等待一部分输出
    print(f"当前: {buffer.buffer}")  # 部分输出

    await task
    print(f"完成: {buffer.buffer}")  # 完整输出

    # 测试中断
    buffer2 = StreamBuffer()
    task2 = asyncio.create_task(simulate_stream("这是一段很长的文本...", buffer2))
    await asyncio.sleep(0.05)
    buffer2.stop()  # 中断
    await task2
    print(f"中断后: {buffer2.buffer}")  # 应只有部分内容

asyncio.run(test_stream())
```
