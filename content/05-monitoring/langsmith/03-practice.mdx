from langsmith.evaluation import evaluate
from langsmith.schemas import Run, Example


# 定义自定义评估函数
def evaluate_response_length(run: Run, example: Example | None = None) -> dict:
    """评估回复长度是否适中（100-500 字符）。"""
    outputs = run.outputs or {}
    response = str(outputs.get("response", ""))
    length = len(response)

    score = 1.0 if 100 <= length <= 500 else 0.0
    return {
        "key": "response_length",
        "score": score,
        "comment": f"Response length: {length} chars",
    }


def evaluate_contains_code(run: Run, example: Example | None = None) -> dict:
    """评估回复是否包含代码块。"""
    outputs = run.outputs or {}
    response = str(outputs.get("response", ""))
    contains_code = "```" in response

    return {
        "key": "contains_code",
        "score": 1.0 if contains_code else 0.0,
