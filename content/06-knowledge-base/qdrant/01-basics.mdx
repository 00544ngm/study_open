## 核心

```python
from qdrant_client import QdrantClient
from qdrant_client.models import PointStruct, Filter, FieldCondition, MatchValue
from openai import OpenAI

class KnowledgeBase:
    def __init__(self):
        self.qdrant = QdrantClient(host="localhost", port=6333)
        self.openai = OpenAI()
        self.collection = "learning_notes"

    def add_document(self, text: str, metadata: dict, point_id: int):
        vector = self.openai.embeddings.create(
            model="text-embedding-3-small",
            input=text
        ).data[0].embedding

        self.qdrant.upsert(
            collection_name=self.collection,
            points=[PointStruct(
                id=point_id,
                vector=vector,
                payload={"text": text, **metadata}
            )]
        )

    def search(self, query: str, stage_filter: str = None, top_k: int = 5):
        query_vec = self.openai.embeddings.create(
            model="text-embedding-3-small",
            input=query
        ).data[0].embedding

        filter_cond = None
        if stage_filter:
            filter_cond = Filter(
                must=[FieldCondition(key="stage", match=MatchValue(value=stage_filter))]
            )

        results = self.qdrant.search(
            collection_name=self.collection,
            query_vector=query_vec,
            query_filter=filter_cond,
            limit=top_k,
            score_threshold=0.5
        )
        return results
```
