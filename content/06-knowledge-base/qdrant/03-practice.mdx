```python
# 使用简单的字符计数模拟向量（实际应使用 Embedding API）
def simple_embed(text: str, dim: int = 384) -> list[float]:
    """简单的模拟向量生成（仅用于练习）。"""
    import hashlib
    vec = []
    for i in range(dim):
        h = hashlib.md5(f"{text}{i}".encode()).hexdigest()
        vec.append(int(h[:8], 16) / 2**32)
    return vec


points = []
for i, doc in enumerate(documents):
    vector = simple_embed(doc["text"])
    point = PointStruct(
        id=i+1,
        vector=vector,
        payload=doc,
    )
    points.append(point)

# 插入到集合
client.upsert(collection_name=collection_name, points=points)
print(f"Inserted {len(points)} points")
```
