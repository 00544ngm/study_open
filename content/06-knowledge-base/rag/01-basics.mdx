## 入门

```text
用户提问
    ↓
1. 向量化问题
    ↓
2. 在向量库中检索相似文档
    ↓
3. 将文档 + 问题组装成 Prompt
    ↓
4. LLM 基于上下文生成回答
```

## 核心

### 完整 RAG 实现

```python
from openai import OpenAI
from typing import List

class RAGSystem:
    def __init__(self):
        self.client = OpenAI()
        self.documents: List[dict] = []  # [{text: str, vector: list}]

    def add_document(self, text: str):
        vector = self.client.embeddings.create(
            model="text-embedding-3-small",
            input=text
        ).data[0].embedding
        self.documents.append({"text": text, "vector": vector})

    def search(self, query: str, top_k: int = 3) -> List[str]:
        query_vec = self.client.embeddings.create(
            model="text-embedding-3-small",
            input=query
        ).data[0].embedding

        scores = []
        for doc in self.documents:
            sim = cosine_similarity(query_vec, doc["vector"])
            scores.append((sim, doc["text"]))

        scores.sort(reverse=True)
        return [text for _, text in scores[:top_k]]

    def generate(self, query: str) -> str:
        contexts = self.search(query)
        prompt = f"""基于以下信息回答问题:

{chr(10).join(f'- {c}' for c in contexts)}

问题: {query}

回答:"""
        response = self.client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[{"role": "user", "content": prompt}]
        )
        return response.choices[0].message.content
```
