### 练习 1：基础 CI 工作流

```yaml
# 补全 GitHub Actions CI 工作流
# .github/workflows/ci.yml

name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  lint-and-test:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Set up Python
      uses: actions/setup-python@v5
      with:
        # 你的代码：设置 Python 3.12 并启用缓存
        python-version: '3.12'
        cache: 'pip'

    - name: Install dependencies
      # 你的代码：安装依赖（requirements.txt 和 dev 依赖）
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
        # 你的代码：安装 pytest、ruff 等开发工具

    - name: Lint with ruff
      # 你的代码：运行 ruff check

    - name: Type check with mypy
      # 你的代码：运行 mypy 检查（如果配置了）

    - name: Run tests
      # 你的代码：运行 pytest，生成覆盖率报告
      run: |
        pytest --cov=app --cov-report=xml

    - name: Upload coverage
```

### 练习 2：CD 部署工作流

```yaml
# 补全部署工作流
# .github/workflows/deploy.yml

name: Deploy

on:
  push:
    branches: [main]
    # 你的代码：只在 backend/ 目录变化时触发
    paths:
      - 'backend/**'

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Configure AWS credentials
      # 你的代码：配置 AWS 凭证（使用 secrets）
      uses: aws-actions/configure-aws-credentials@v4
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: us-east-1

    - name: Login to Amazon ECR
      # 你的代码：登录到 ECR
      id: login-ecr
      uses: aws-actions/amazon-ecr-login@v2

    - name: Build and push Docker image
      # 你的代码：构建 Docker 镜像并推送到 ECR
      env:
        ECR_REGISTRY: ${{ steps.login-ecr.outputs.registry }}
        ECR_REPOSITORY: agent-api
        IMAGE_TAG: ${{ github.sha }}
      run: |
        docker build -t $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG ./backend
        docker push $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG

    - name: Deploy to ECS
      # 你的代码：更新 ECS 服务
      run: |
        aws ecs update-service --cluster agent-cluster \
          --service agent-api --force-new-deployment
```

### 练习 3：工作流矩阵策略

```yaml
# 使用矩阵策略测试多 Python 版本

name: Matrix Test

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    # 你的代码：配置矩阵策略，测试 Python 3.11、3.12、3.13
    strategy:
      matrix:
        python-version: ['3.11', '3.12', '3.13']

    steps:
    - uses: actions/checkout@v4

    - name: Set up Python
      uses: actions/setup-python@v5
      with:
        python-version: ${{ matrix.python-version }}
        cache: 'pip'

    - name: Install dependencies
      run: pip install -r requirements.txt

    - name: Run tests
      run: pytest

    # 你的代码：添加一个通知步骤
    # 当测试失败时，发送 Slack 通知（使用 rtCamp/action-slack-notify@v2）
    # 提示：使用 if: failure() 条件
```
