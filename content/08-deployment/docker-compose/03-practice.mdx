### 练习 2：服务健康检查

```yaml
# 为 postgres 和 api 服务添加健康检查

version: "3.8"

services:
  postgres:
    image: postgres:16-alpine
    environment:
      POSTGRES_DB: agent
      POSTGRES_USER: agent
      POSTGRES_PASSWORD: agent_pass
    # 你的代码：添加 healthcheck
    # 使用 pg_isready -U agent 检查
    # 间隔 10s，超时 5s，重试 5 次
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U agent"]
      interval: 10s
      timeout: 5s
      retries: 5

  api:
    build: ./backend
    ports: ["8000:8000"]
    environment:
      DATABASE_URL: postgresql://agent:agent_pass@postgres:5432/agent
    # 你的代码：添加 healthcheck
    # 使用 curl http://localhost:8000/health 检查
    # 添加 condition: service_healthy 到 depends_on
    depends_on:
      postgres:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
```
