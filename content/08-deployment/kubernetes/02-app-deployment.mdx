## 完整应用部署

本节将一个 AI Agent 应用完整部署到 Kubernetes 集群，涵盖 Deployment、Service、ConfigMap、Secrets、Ingress、HPA 等核心资源。

## Deployment & Service

### API 服务部署

```yaml
# api-deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: agent-api
  namespace: agent-platform
  labels:
    app: agent-api
    tier: backend
    version: v1
spec:
  replicas: 3
  # 更新策略
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1           # 更新时最多额外启动 1 个 Pod
      maxUnavailable: 0     # 更新时不允许不可用
  selector:
    matchLabels:
      app: agent-api
  template:
    metadata:
      labels:
        app: agent-api
        version: v1
    spec:
      # Pod 安全上下文
      securityContext:
        runAsNonRoot: true
        runAsUser: 1000
        fsGroup: 2000

      containers:
      - name: agent-api
        image: registry.example.com/agent-api:v1.2.3
        imagePullPolicy: IfNotPresent
        ports:
        - containerPort: 8000
          protocol: TCP
          name: http
        - containerPort: 9090
          protocol: TCP
          name: metrics

        # 从 ConfigMap 注入环境变量
        envFrom:
        - configMapRef:
            name: agent-config
        # 从 Secret 注入敏感信息
        - secretRef:
            name: app-secrets

        # 单个环境变量
        env:
        - name: POD_NAME
          valueFrom:
            fieldRef:
              fieldPath: metadata.name
        - name: NAMESPACE
          valueFrom:
            fieldRef:
              fieldPath: metadata.namespace
        - name: DB_PASSWORD
          valueFrom:
            secretKeyRef:
              name: db-secret
              key: password

        # 资源限制
        resources:
          requests:
            cpu: 500m
            memory: 512Mi
          limits:
            cpu: 1000m
            memory: 1Gi

        # 存活探针（容器是否正常）
        livenessProbe:
          httpGet:
            path: /healthz
            port: 8000
          initialDelaySeconds: 30
          periodSeconds: 15
          timeoutSeconds: 5
          failureThreshold: 3

        # 就绪探针（Pod 是否可接收流量）
        readinessProbe:
          httpGet:
            path: /readyz
            port: 8000
          initialDelaySeconds: 15
          periodSeconds: 10
          timeoutSeconds: 3
          successThreshold: 1
          failureThreshold: 2

        # 启动探针（慢启动应用）
        startupProbe:
          httpGet:
            path: /healthz
            port: 8000
          initialDelaySeconds: 10
          periodSeconds: 5
          failureThreshold: 30  # 容许 30 次失败（150 秒启动时间）

        # 挂载卷
        volumeMounts:
        - name: tmp
          mountPath: /tmp
        - name: app-config
          mountPath: /app/config
          readOnly: true

      # 优雅终止
      terminationGracePeriodSeconds: 60

      # 镜像拉取秘钥
      imagePullSecrets:
      - name: registry-credentials

      volumes:
      - name: tmp
        emptyDir: {}
      - name: app-config
        configMap:
          name: app-config-files
---
# api-service.yaml
apiVersion: v1
kind: Service
metadata:
  name: agent-api-service
  namespace: agent-platform
  labels:
    app: agent-api
  annotations:
    prometheus.io/scrape: "true"
    prometheus.io/port: "9090"
spec:
  selector:
    app: agent-api
  ports:
  - name: http
    port: 80
    targetPort: 8000
  - name: metrics
    port: 9090
    targetPort: 9090
  type: ClusterIP
  # 会话保持（如需）
  # sessionAffinity: ClientIP
```

## Ingress

Ingress 将外部流量路由到集群内的 Service。

```yaml
# ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: agent-api-ingress
  namespace: agent-platform
  annotations:
    # 使用 nginx-ingress 控制器
    kubernetes.io/ingress.class: "nginx"
    nginx.ingress.kubernetes.io/ssl-redirect: "true"
    nginx.ingress.kubernetes.io/force-ssl-redirect: "true"
    # 跨域配置
    nginx.ingress.kubernetes.io/enable-cors: "true"
    nginx.ingress.kubernetes.io/cors-allow-origin: "https://app.agent.example.com"
    nginx.ingress.kubernetes.io/cors-allow-methods: "GET, POST, PUT, DELETE, OPTIONS"
    nginx.ingress.kubernetes.io/cors-allow-credentials: "true"
    # 请求体大小
    nginx.ingress.kubernetes.io/proxy-body-size: "10m"
    # 超时
    nginx.ingress.kubernetes.io/proxy-read-timeout: "60s"
    nginx.ingress.kubernetes.io/proxy-connect-timeout: "10s"
    # 速率限制
    nginx.ingress.kubernetes.io/limit-rps: "50"
    nginx.ingress.kubernetes.io/limit-burst-multiplier: "3"
    # 客户端真实 IP
    nginx.ingress.kubernetes.io/forwarded-for-header: "X-Forwarded-For"
    # 安全头部
    nginx.ingress.kubernetes.io/configuration-snippet: |
      add_header X-Frame-Options "SAMEORIGIN" always;
      add_header X-Content-Type-Options "nosniff" always;
      add_header X-XSS-Protection "1; mode=block" always;
spec:
  tls:
  - hosts:
    - api.agent.example.com
    - ws.agent.example.com
    secretName: agent-tls
  rules:
  - host: api.agent.example.com
    http:
      paths:
      - path: /api/v1
        pathType: Prefix
        backend:
          service:
            name: agent-api-service
            port:
              number: 80
      - path: /ws
        pathType: Prefix
        backend:
          service:
            name: agent-ws-service
            port:
              number: 8001
  - host: admin.agent.example.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: agent-admin-service
            port:
              number: 80
```

## Horizontal Pod Autoscaler

基于 CPU/内存或自定义指标自动扩缩容。

### 基于 CPU 和内存

```yaml
# hpa.yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: agent-api-hpa
  namespace: agent-platform
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: agent-api
  minReplicas: 2
  maxReplicas: 20
  behavior:
    scaleDown:
      stabilizationWindowSeconds: 300     # 缩容稳定窗口
      policies:
      - type: Percent
        value: 50
        periodSeconds: 60
    scaleUp:
      stabilizationWindowSeconds: 0       # 立即扩容
      policies:
      - type: Percent
        value: 100                         # 每次翻倍
        periodSeconds: 15
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 70
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80
```

### 基于自定义指标（每秒请求数）

```yaml
# 需要 Prometheus Adapter 或 KEDA
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: agent-api-rps-hpa
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: agent-api
  minReplicas: 2
  maxReplicas: 50
  metrics:
  - type: Pods
    pods:
      metric:
        name: http_requests_per_second
      target:
        type: AverageValue
        averageValue: 100
```

```bash
# 查看 HPA 状态
kubectl get hpa -n agent-platform
kubectl describe hpa agent-api-hpa -n agent-platform

# 模拟负载测试自动扩缩容
kubectl run load-generator --image=busybox -- /bin/sh -c \
  "while true; do wget -q -O- http://agent-api-service.agent-platform.svc.cluster.local/health; done"

# 资源监控
kubectl top pods -n agent-platform
kubectl top nodes
```

## 存活/就绪/启动探针

三种探针的适用场景：

| 探针 | 作用 | 失败后果 | 推荐场景 |
|------|------|----------|----------|
| `livenessProbe` | 检查容器是否存活 | 重启 Pod | 死锁、内存泄漏 |
| `readinessProbe` | 检查服务是否就绪 | 从 Service 移除 | 启动加载、依赖未就绪 |
| `startupProbe` | 检查应用是否启动完成 | 重启 Pod | 慢启动应用（> 启动探针周期前不执行存活探针） |

### 自定义探针类型

```yaml
# TCP 探针（数据库等服务）
livenessProbe:
  tcpSocket:
    port: 5432
  initialDelaySeconds: 15
  periodSeconds: 20

# 命令探针（Redis）
livenessProbe:
  exec:
    command:
    - redis-cli
    - ping
  initialDelaySeconds: 15
  periodSeconds: 10

# gRPC 探针（K8s 1.24+）
livenessProbe:
  grpc:
    port: 50051
  initialDelaySeconds: 15
```

### gRPC 健康检查实现

```go
// 服务端示例（Go）
import "google.golang.org/grpc/health/grpc_health_v1"

func (s *server) Check(ctx context.Context, req *grpc_health_v1.HealthCheckRequest) (*grpc_health_v1.HealthCheckResponse, error) {
    if s.db == nil {
        return &grpc_health_v1.HealthCheckResponse{
            Status: grpc_health_v1.HealthCheckResponse_NOT_SERVING,
        }, nil
    }
    return &grpc_health_v1.HealthCheckResponse{
        Status: grpc_health_v1.HealthCheckResponse_SERVING,
    }, nil
}
```

## Helm 基础

Helm 是 K8s 的包管理器，使用 Chart 管理应用定义。

### Chart 结构

```

├── Chart.yaml              # Chart 元信息
├── values.yaml             # 默认配置值
├── values-production.yaml  # 生产环境覆盖
├── templates/
│   ├── _helpers.tpl        # 模板帮助函数

│   ├── hpa.yaml            # HPA 模板

```

## 完整部署流程

```bash
# 1. 创建命名空间
kubectl create namespace agent-platform

# 2. 部署基础设施
kubectl apply -f configmap.yaml -n agent-platform
kubectl apply -f secrets.yaml -n agent-platform

# 3. 部署数据服务（使用 Helm）
helm repo add bitnami https://charts.bitnami.com/bitnami
helm upgrade --install postgresql bitnami/postgresql \
  --namespace agent-platform \
  --set auth.database=agent_platform \
  --set auth.username=agent \
  --set auth.password=strong-password \
  --set primary.persistence.size=50Gi

helm upgrade --install redis bitnami/redis \
  --namespace agent-platform \
  --set architecture=standalone \
  --set auth.enabled=false

# 4. 部署应用
kubectl apply -f api-deployment.yaml -n agent-platform
kubectl apply -f api-service.yaml -n agent-platform
kubectl apply -f worker-deployment.yaml -n agent-platform

# 5. 配置自动扩缩容
kubectl apply -f hpa.yaml -n agent-platform

# 6. 配置 Ingress
kubectl apply -f ingress.yaml -n agent-platform

# 7. 验证
kubectl get all -n agent-platform
kubectl get ingress -n agent-platform
curl -H "Host: api.agent.example.com" http://<INGRESS_IP>/api/v1/health
```
