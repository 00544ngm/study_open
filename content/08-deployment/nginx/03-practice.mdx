## 练习

### 目标

通过动手练习掌握 Nginx 的配置，包括反向代理、负载均衡、SSL 配置和限流。

### 练习 1：反向代理配置

```nginx
# 补全 Nginx 反向代理配置
# 将 api.example.com 的请求代理到本地的 Agent API 服务

server {
    # 你的代码：监听 80 端口
    # 你的代码：设置 server_name 为 api.example.com

    location / {
        # 你的代码：代理到 http://localhost:8000
        proxy_pass http://localhost:8000;

        # 你的代码：设置 Host 头为原始请求的 Host
        proxy_set_header Host $host;

        # 你的代码：设置 X-Real-IP 头为客户端真实 IP
        proxy_set_header X-Real-IP $remote_addr;

        # 你的代码：添加 CORS 头允许跨域
        add_header Access-Control-Allow-Origin "*";
        add_header Access-Control-Allow-Methods "GET, POST, OPTIONS";
        add_header Access-Control-Allow-Headers "Authorization, Content-Type";
    }

    # 你的代码：添加 /health 路径的特殊处理
    # 直接返回 200，不做代理（用于健康检查）
    location /health {
        return 200 "OK";
    }
}
```

### 练习 2：负载均衡与限流

```nginx
# 配置负载均衡和限流

# 1. 定义上游服务器组
upstream agent_api {
    # 你的代码：添加 3 台后端服务器
    # server api1:8000 weight=3（权重 3）
    # server api2:8000 weight=2
    # server api3:8000 weight=1
    server api1:8000 weight=3;
    server api2:8000 weight=2;
    server api3:8000 weight=1;

    # 你的代码：添加健康检查（nginx Plus 需要，这里用被动检查）
    # max_fails=3 fail_timeout=30s
}

# 2. 定义限流区域
# 你的代码：创建限流区域，名称为 "api_limit"
# 使用 10MB 共享内存，限制每秒 10 个请求
limit_req_zone $binary_remote_addr zone=api_limit:10m rate=10r/s;

server {
    listen 80;
    server_name api.example.com;

    # 3. 应用限流
    location /api/ {
        # 你的代码：应用限流规则，超过限制时返回 429
        limit_req zone=api_limit burst=20 nodelay;

        # 你的代码：代理到 upstream agent_api
        proxy_pass http://agent_api;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # 4. 静态文件直接由 Nginx 服务
    location /static/ {
        # 你的代码：设置静态文件根目录为 /var/www/static
        alias /var/www/static/;
        # 你的代码：设置缓存过期时间为 30 天
        expires 30d;
    }
}
```

### 练习 3：SSL 与安全配置

```nginx
# 为 Agent 平台配置 HTTPS

# HTTP → HTTPS 重定向
server {
    listen 80;
    server_name agent.example.com;
    # 你的代码：301 重定向到 https://agent.example.com$request_uri
    return 301 https://$host$request_uri;
}

# HTTPS 服务器
server {
    # 你的代码：监听 443 ssl
    listen 443 ssl;
    server_name agent.example.com;

    # SSL 证书配置
    # 你的代码：设置证书路径
    ssl_certificate /etc/ssl/certs/agent.example.com.pem;
    ssl_certificate_key /etc/ssl/private/agent.example.com.key;

    # SSL 协议和加密套件
    # 你的代码：只允许 TLS 1.2 和 1.3
    ssl_protocols TLSv1.2 TLSv1.3;
    # 你的代码：设置安全的加密套件
    ssl_ciphers HIGH:!aNULL:!MD5;

    # 安全头
    # 你的代码：添加以下安全头
    # X-Content-Type-Options: nosniff（防止 MIME 嗅探）
    # X-Frame-Options: DENY（防止点击劫持）
    # X-XSS-Protection: 1; mode=block（XSS 过滤）

    location / {
        proxy_pass http://agent_api;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        # 你的代码：传递协议信息给后端
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
```
