        if not data:
            return 0.0

    def percentile(self, metric_name: str, p: float, duration_minutes: int = 60) -> float:
        """计算百分位值。"""
        data = self.query(metric_name, duration_minutes)
        if not data:
            return 0.0
        values = sorted(d["value"] for d in data)
        idx = int(len(values) * p / 100)
