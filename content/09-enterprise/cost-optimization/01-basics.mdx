## 入门

LLM API 按 Token 计费，如果不加管理，成本很容易失控。

## 核心

### Token 计数器

```python
import tiktoken

def count_tokens(text: str, model: str = "gpt-4o") -> int:
    """统计文本的 Token 数"""
    try:
        encoding = tiktoken.encoding_for_model(model)
    except:
        encoding = tiktoken.get_encoding("cl100k_base")
    return len(encoding.encode(text))

def estimate_cost(tokens: int, model: str = "gpt-4o") -> float:
    """估算成本（美元）"""
    pricing = {
        "gpt-4o": {"input": 2.50 / 1_000_000, "output": 10.00 / 1_000_000},
        "gpt-4o-mini": {"input": 0.15 / 1_000_000, "output": 0.60 / 1_000_000},
        "claude-sonnet-4-20250514": {"input": 3.00 / 1_000_000, "output": 15.00 / 1_000_000},
    }
    if model not in pricing:
        return 0

    # 假设输出约为输入的一半
    input_cost = tokens * pricing[model]["input"]
    output_cost = (tokens // 2) * pricing[model]["output"]
    return input_cost + output_cost

# 使用
text = "介绍 AI Agent 的核心概念..." * 100
tokens = count_tokens(text)
cost = estimate_cost(tokens, "gpt-4o")
print(f"Token: {tokens}, 预估成本: ${cost:.4f}")
```

### 成本追踪

```python
import json
from datetime import datetime

class CostTracker:
    def __init__(self, log_file="cost_log.json"):
        self.log_file = log_file
        self.sessions = []

    def log_call(self, model: str, input_tokens: int, output_tokens: int):
        entry = {
            "timestamp": datetime.now().isoformat(),
            "model": model,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "cost": self._calc_cost(model, input_tokens, output_tokens)
        }
        self.sessions.append(entry)
        self._persist()

    def _calc_cost(self, model, input_tokens, output_tokens):
        pricing = {
            "gpt-4o": (2.50, 10.00),
            "gpt-4o-mini": (0.15, 0.60),
        }
        input_rate, output_rate = pricing.get(model, (0, 0))
        return (input_tokens * input_rate + output_tokens * output_rate) / 1_000_000

    def daily_summary(self) -> dict:
        today = datetime.now().strftime("%Y-%m-%d")
        today_calls = [
            s for s in self.sessions
            if s["timestamp"].startswith(today)
        ]
        total_cost = sum(s["cost"] for s in today_calls)
        total_input = sum(s["input_tokens"] for s in today_calls)
        total_output = sum(s["output_tokens"] for s in today_calls)

        return {
            "date": today,
            "calls": len(today_calls),
            "total_cost": round(total_cost, 4),
            "input_tokens": total_input,
            "output_tokens": total_output,
        }

    def _persist(self):
        with open(self.log_file, "w") as f:
            json.dump(self.sessions, f)
```

### 缓存策略

```python
import hashlib
import json
import redis.asyncio as redis

class LLMCache:
    """LLM 响应缓存：相同问题直接返回缓存结果"""

    def __init__(self, redis_client):
        self.redis = redis_client
        self.hits = 0
        self.misses = 0

    def _make_key(self, messages: list, model: str) -> str:
        content = json.dumps(messages, sort_keys=True) + model
        return f"llm_cache:{hashlib.md5(content.encode()).hexdigest()}"

    async def get_or_compute(self, messages, model, llm_func):
        key = self._make_key(messages, model)

        cached = await self.redis.get(key)
        if cached:
            self.hits += 1
            return json.loads(cached)

        self.misses += 1
        result = await llm_func(messages, model)

        # 缓存 1 小时
        await self.redis.setex(key, 3600, json.dumps(result))
        return result

    @property
    def hit_rate(self) -> float:
        total = self.hits + self.misses
        return self.hits / total if total > 0 else 0
```

### 模型选型策略

```python
class ModelRouter:
    """根据任务复杂度选择合适的模型"""

    def __init__(self):
        self.models = {
            "cheap": "gpt-4o-mini",     # 简单任务
            "balanced": "gpt-4o",        # 一般任务
            "powerful": "claude-sonnet-4-20250514",  # 复杂任务
        }

    def select_model(self, task: str, input_length: int) -> str:
        # 短文本 + 简单任务 → 便宜模型
        if input_length < 500 and len(task) < 50:
            return self.models["cheap"]

        # 长文本或复杂推理 → 强模型
        if input_length > 4000 or any(kw in task for kw in ["分析", "总结", "翻译", "代码"]):
            return self.models["powerful"]

        return self.models["balanced"]
```
