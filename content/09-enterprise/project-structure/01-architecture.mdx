### 配置管理

```python
# config/config.py
from pydantic_settings import BaseSettings
from typing import Optional

class Settings(BaseSettings):
    # 应用
    app_name: str = "AI Agent"
    debug: bool = False
    log_level: str = "INFO"

    # LLM
    openai_api_key: Optional[str] = None
    anthropic_api_key: Optional[str] = None
    default_model: str = "gpt-4o"
    max_tokens: int = 4096
    temperature: float = 0.7

    # 外部服务
    redis_url: str = "redis://localhost:6379"
    qdrant_url: str = "http://localhost:6333"
    database_url: str = "sqlite:///app.db"

    # Agent
    agent_timeout: int = 30
    max_retries: int = 3

    class Config:
        env_file = ".env"
        env_file_encoding = "utf-8"

settings = Settings()
```

## 进阶

### 分层架构

```python
# 清晰的职责分层
"""
Router (API层)     → 请求/响应格式，认证，限流
    ↓
Service (业务层)   → Agent 编排，工作流
    ↓
Agent (Agent层)    → LLM 调用，工具选择
    ↓
Tool (工具层)      → 具体工具实现
    ↓
External (外部服务) → Redis, Qdrant, Database
"""

# 示例：Service 层
class ChatService:
    def __init__(self):
        self.agent = CustomerServiceAgent()
        self.memory = RedisMemory()

    async def process_message(self, user_id: str, message: str) -> str:
        history = await self.memory.get_history(user_id)
        response = await self.agent.run(message, history)
        await self.memory.save(user_id, message, response)
        return response
```

### 依赖注入

```python
from functools import lru_cache

@lru_cache()
def get_settings() -> Settings:
    return Settings()

def get_llm_client():
    return LLMFactory.create(settings=get_settings())

def get_vector_store():
    return QdrantClient(url=get_settings().qdrant_url)

# FastAPI 中使用
@app.post("/chat")
async def chat(
    request: ChatRequest,
    llm: OpenAI = Depends(get_llm_client),
    store: QdrantClient = Depends(get_vector_store),
):
    pass
```
