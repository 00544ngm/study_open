import asyncio
from typing import Any, Callable, List
from dataclasses import dataclass, field


@dataclass
class ParallelTask:
    name: str
    handler: Callable
    args: tuple = ()
    result: Any = None
    error: str | None = None


class ParallelWorkflow:
    """并行模式：多个任务同时执行，等待所有完成"""

    def __init__(self, max_concurrency: int = 5):
        self.max_concurrency = max_concurrency

    async def run_task(self, task: ParallelTask) -> ParallelTask:

            else:

class AggregateWorkflow(ParallelWorkflow):
    """并行 + 聚合模式"""

    def __init__(self, aggregator: Callable[[List[Any]], Any], max_concurrency: int = 5):
        super().__init__(max_concurrency)
        self.aggregator = aggregator

    async def execute_and_aggregate(self, tasks: List[ParallelTask]) -> Any:
        """
        并行执行任务，然后聚合结果。
        过滤掉出错的任务，只聚合成功的。
        """
        results = await self.execute(tasks)
        successful = [t.result for t in results if t.error is None]
        if successful:
            return self.aggregator(successful)
        return "所有任务失败"
