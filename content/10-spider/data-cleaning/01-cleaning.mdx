## 处理缺失值

爬虫经常遇到缺失字段：

```python
# 检测缺失
df.isnull().sum()
df.isna().sum()

# 删除含缺失的行

# 删除指定列有缺失的行
df_clean = df.dropna(subset=["title", "url"])

# 填充缺失

df["description"].fillna("", inplace=True)

# 用均值/中位数填充
df["price"].fillna(df["price"].median(), inplace=True)

# 前向填充（时间序列常用）
df.sort_values("date", inplace=True)
df["price"].fillna(method="ffill", inplace=True)
```

## 清洗实战管线

```python
import pandas as pd
import re

def clean_scraped_data(df: pd.DataFrame) -> pd.DataFrame:
    """完整的数据清洗管线"""

    # 1. 复制以防修改原始数据
    df = df.copy()

    # 2. 基本信息记录
    print(f"原始行数: {len(df)}")
    print(f"原始列数: {len(df.columns)}")

    # 3. 删除完全空的行列
    df.dropna(how="all", inplace=True)

    # 4. 去重（按 URL 去重）
    if "url" in df.columns:
        before = len(df)
        df.drop_duplicates(subset=["url"], keep="first", inplace=True)
        print(f"去重移除: {before - len(df)} 行")

    # 5. 移除 URL 为空的行
    if "url" in df.columns:
        df = df[df["url"].notna() & (df["url"] != "")]

    # 6. 清洗标题
    if "title" in df.columns:
        df["title"] = (
            df["title"]
            .str.strip()
            .str.replace(r"\s+", " ", regex=True)
        )
        # 删除标题过短的行
        df = df[df["title"].str.len() >= 2]

    # 7. 标准化日期
    if "date" in df.columns:
        df["date"] = pd.to_datetime(df["date"], errors="coerce")
        # 删除日期解析失败的行
        df = df[df["date"].notna()]

    # 8. 填充缺失
    if "description" in df.columns:
        df["description"] = df["description"].fillna("")
    if "views" in df.columns:
        df["views"] = (
            pd.to_numeric(df["views"], errors="coerce")
            .fillna(0)
            .astype(int)
        )

    # 9. 重置索引
    df.reset_index(drop=True, inplace=True)

    print(f"清洗后行数: {len(df)}")
    return df

# 使用
raw_df = pd.read_csv("scraped_articles.csv")
clean_df = clean_scraped_data(raw_df)
clean_df.to_csv("cleaned_articles.csv", index=False)
```

## 异常值处理

```python
import numpy as np

# Z-score 方法
def remove_outliers_zscore(df, column, threshold=3):
    z_scores = np.abs(
        (df[column] - df[column].mean()) / df[column].std()
    )
    return df[z_scores < threshold]

# IQR 方法
def remove_outliers_iqr(df, column):
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    return df[
        (df[column] >= Q1 - 1.5 * IQR) &
        (df[column] <= Q3 + 1.5 * IQR)
    ]

# 价格异常检测
df_clean = remove_outliers_iqr(df, "price")
```
