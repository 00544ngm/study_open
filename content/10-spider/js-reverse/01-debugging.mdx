## 使用 AST 还原

AST（抽象语法树）可以精确分析和修改 JS 代码：

```javascript
// 安装: npm install @babel/parser @babel/generator @babel/traverse

const parser = require('@babel/parser');
const generate = require('@babel/generator').default;
const traverse = require('@babel/traverse').default;

// 解析为 AST
const ast = parser.parse(obfuscatedCode);

// 遍历并修改
traverse(ast, {
    // 还原字符串拼接
    BinaryExpression(path) {
        if (path.node.operator === '+') {
            const { left, right } = path.node;
            if (left.type === 'StringLiteral' && right.type === 'StringLiteral') {
                path.replaceWith({
                    type: 'StringLiteral',
                    value: left.value + right.value,
                });
            }
        }
    },
    // 还原数字进制
    NumericLiteral(path) {
        // 0x1a → 26
        path.node.extra = undefined;
    },
});

// 重新生成
const { code } = generate(ast);
console.log(code);
```

## Hook 技术

Hook 是在函数执行前拦截并修改行为：

```javascript
// 在 DevTools Console 中执行

// 1. Hook JSON.parse（查看解密后的数据结构）
const originalParse = JSON.parse;
JSON.parse = function(text) {
    console.log("[Hook] JSON.parse:", text.slice(0, 200));
    debugger;  // 暂停查看调用栈
    return originalParse(text);
};

// 2. Hook atob（Base64 解密）
const originalAtob = window.atob;
window.atob = function(str) {
    console.log("[Hook] atob:", str);
    debugger;
    return originalAtob(str);
};

// 3. Hook Cookie 设置
Object.defineProperty(document, 'cookie', {
    set: function(val) {
        console.log("[Hook] set cookie:", val);
        debugger;
        return val;
    },
});
```

## Python 调用 JS

找到加密算法后用 Python 执行：

```python
# 方式 1：PyExecJS
import execjs

# 加载 JS 文件
with open("encrypt.js") as f:
    js_code = f.read()

ctx = execjs.compile(js_code)
result = ctx.call("encrypt", "hello")
print(result)

# 方式 2：Node.js 子进程
import subprocess, json

js_code = """
const crypto = require('crypto');
const data = process.argv[1];
const hash = crypto.createHash('md5').update(data).digest('hex');
console.log(JSON.stringify({hash}));
"""
result = subprocess.run(
    ["node", "-e", js_code, "hello"],
    capture_output=True, text=True,
)
print(json.loads(result.stdout))

# 方式 3：PyMiniRacer (更快)
from py_mini_racer import MiniRacer
ctx = MiniRacer()
ctx.eval(js_code)
result = ctx.call("encrypt", "hello")
```
