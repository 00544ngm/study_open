### 代码关系图构建

```python
import ast

class DependencyGraph:
    """构建文件的 import/export 依赖关系图。"""

    def __init__(self):
        self.graph: dict[str, set[str]] = {}

    def analyze_file(self, path: str):
        tree = ast.parse(open(path).read())
        imports = set()

        for node in ast.walk(tree):
            if isinstance(node, ast.Import):
                for alias in node.names:
                    imports.add(alias.name)
            elif isinstance(node, ast.ImportFrom):
                if node.module:
                    imports.add(node.module)

        rel_path = path.replace("\\", "/")
        self.graph[rel_path] = imports

    def find_affected_files(self, changed_file: str) -> list[str]:
        """找到受该文件变更影响的所有文件。"""
        affected = []
        for file_path, imports in self.graph.items():
            if changed_file in imports:
                affected.append(file_path)
        return affected

    def topological_sort(self) -> list[str]:
        """按依赖关系拓扑排序。"""
        visited = set()
        result = []

        def dfs(node: str):
            if node in visited:
                return
            visited.add(node)
            for dep in self.graph.get(node, set()):
                dfs(dep)
            result.append(node)

        for node in self.graph:
            dfs(node)

        return result
```
