### 精确的行级编辑

```python
from difflib import unified_diff

def apply_edit(path: str, old_str: str, new_str: str) -> bool:
    """基于字符串匹配的精确替换。"""
    with open(path, "r") as f:
        content = f.read()

    if old_str not in content:
        return False  # 匹配失败，不做修改

    content = content.replace(old_str, new_str, 1)

    with open(path, "w") as f:
        f.write(content)

    return True


def search_replace_edit(path: str, search: str, replace: str):
    """带搜索上下文的精确替换，减少误匹配。"""
    with open(path, "r") as f:
        lines = f.readlines()

    search_lines = search.split("\n")
    replace_lines = replace.split("\n")

    for i in range(len(lines) - len(search_lines) + 1):
        if lines[i:i + len(search_lines)] == [l + "\n" for l in search_lines]:
            lines[i:i + len(search_lines)] = [l + "\n" for l in replace_lines]
            break

    with open(path, "w") as f:
        f.writelines(lines)
```

### 备份与回滚

```python
import shutil
from datetime import datetime

class FileBackup:
    def __init__(self, base_dir: str):
        self.base_dir = base_dir
        self.backup_dir = f"{base_dir}/.agent_backups"

    def backup(self, path: str) -> str:
        """编辑前创建备份，返回备份路径。"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        rel_path = path.replace(self.base_dir, "").strip("/")
        backup_path = f"{self.backup_dir}/{timestamp}/{rel_path}"

        os.makedirs(os.path.dirname(backup_path), exist_ok=True)
        shutil.copy2(path, backup_path)
        return backup_path

    def rollback(self, backup_path: str, original_path: str):
        """从备份恢复文件。"""
        shutil.copy2(backup_path, original_path)

    def rollback_all_before(self, timestamp: str):
        """回滚指定时间戳之前的所有修改。"""
        backup_root = f"{self.backup_dir}/{timestamp}"
        if os.path.exists(backup_root):
            for root, _, files in os.walk(backup_root):
                for f in files:
                    rel = os.path.relpath(root, backup_root)
                    original = f"{self.base_dir}/{rel}/{f}"
                    shutil.copy2(os.path.join(root, f), original)
```
